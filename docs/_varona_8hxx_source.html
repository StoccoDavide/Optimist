120 bool solve_impl(FunctionWrapper function, FirstDerivativeWrapper first_derivative, Real x_ini,
121 Real &x_sol)
122 {
123 #define CMD "Optimist::ScalarRootFinder::Varona::solve(...): "
124
125 // Setup internal variables
126 this->reset();
127
128 // Print header
129 if (this->m_verbose) {this->header();}
130
131 // Initialize variables
132 bool damped;
133 Real residuals, step_norm;
134 Real x_old, x_new, function_old, function_new, step_old, step_new;
135 Real first_derivative_old;
136
137 // Set initial iteration
138 x_old = x_ini;
139 this->evaluate_function(function, x_old, function_old);
140
141 // Algorithm iterations
142 Real tolerance_residuals{this->m_tolerance};
143 Real tolerance_step_norm{this->m_tolerance * this->m_tolerance};
144 for (this->m_iterations = Integer(1); this->m_iterations < this->m_max_iterations; ++this->m_iterations)
145 {
146 // Store trace
147 this->store_trace(x_old);
148
149 // Evaluate first derivative
150 this->evaluate_first_derivative(first_derivative, x_old, first_derivative_old);
151
152 // Calculate step
153 if (std::abs(first_derivative_old) < EPSILON_LOW) {
154 OPTIMIST_WARNING( CMD "singular first derivative detected.");
155 first_derivative_old = (first_derivative_old > Real(0.0)) ? EPSILON_LOW : -EPSILON_LOW;
156 }
157
158 this->compute_step(function, x_old, function_old, first_derivative_old, step_old);
159
160 // Check convergence
161 residuals = std::abs(function_old);
162 step_norm = std::abs(step_old);
163 if (this->m_verbose) {this->info(residuals);}
164 if (residuals < tolerance_residuals || step_norm < tolerance_step_norm) {
165 this->m_converged = true;
166 break;
167 }
168
169 if (this->m_damped) {
170 // Relax the iteration process
171 damped = this->damp(function, x_old, function_old, step_old, x_new, function_new, step_new);
172 OPTIMIST_ASSERT_WARNING(damped, CMD "damping failed.");
173 } else {
174 // Update point
175 x_new = x_old + step_old;
176 this->evaluate_function(function, x_new, function_new);
177 }
178
179 // Update internal variables
180 x_old = x_new;
181 function_old = function_new;
182 step_old = step_new;
183 }
184
185 // Print bottom
186 if (this->m_verbose) {this->bottom();}
187
188 // Convergence data
189 x_sol = x_old;
190 return this->m_converged;
191
192 #undef CMD
193 }

205 void compute_step(FunctionWrapper function, Real x_old, Real function_old, Real first_derivative_old,
206 Real & step_old)
207 {
208 Real function_y, function_z, function_w, function_h, step_tmp, t, s, u, v;
209 Real tolerance_residuals{this->m_tolerance};
210 //Real tolerance_step_norm{this->m_tolerance * this->m_tolerance};
211
212 // Base step
213 step_old = -function_old/first_derivative_old;
214
215 // Order 4 step
216 if (this->m_method == Method::ORDER_4 || this->m_method == Method::ORDER_8 ||
217 this->m_method == Method::ORDER_16 || this->m_method == Method::ORDER_32) {
218 this->evaluate_function(function, x_old+step_old, function_y);
219 if (std::abs(function_y) < tolerance_residuals) {return;}
220 t = function_y/function_old;
221 step_tmp = this->Q(t) * (function_y/first_derivative_old);
222 if (std::isfinite(step_tmp)) {step_old -= step_tmp;}
223 else {return;}
224 }
225
226 // Order 8 step (continued order 4)
227 if (this->m_method == Method::ORDER_8 || this->m_method == Method::ORDER_16 ||
228 this->m_method == Method::ORDER_32) {
229 this->evaluate_function(function, x_old+step_old, function_z);
230 if (std::abs(function_z) < tolerance_residuals) {return;}
231 s = function_z/function_y;
232 step_tmp = this->W(t, s) * (function_z/first_derivative_old);
233 if (std::isfinite(step_tmp)) {step_old -= step_tmp;}
234 else {return;}
235 }
236
237 // Order 16 step (continued order 8)
238 if (this->m_method == Method::ORDER_16 || this->m_method == Method::ORDER_32) {
239 this->evaluate_function(function, x_old+step_old, function_w);
240 if (std::abs(function_w) < tolerance_residuals) {return;}
241 u = function_w/function_z;
242 step_tmp = this->H(t, s, u) * (function_w/first_derivative_old);
243 if (std::isfinite(step_tmp)) {step_old -= step_tmp;}
244 else {return;}
245 }
246
247 // Order 32 step (continued order 16)
248 if (this->m_method == Method::ORDER_32) {
249 this->evaluate_function(function, x_old+step_old, function_h);
250 if (std::abs(function_h) < tolerance_residuals) {return;}
251 v = (function_h/function_w);
252 step_tmp = this->J(t, s, u, v) * (function_h/first_derivative_old);
253 if (std::isfinite(step_tmp)) {step_old -= step_tmp;}
254 else {return;}
255 }
256
257 OPTIMIST_ASSERT(std::isfinite(step_old),
258 "Optimist::ScalarRootFinder::Varona::compute_step(...): step is not finite.");
259 }

274 static Real W(Real t, Real s) {return t*t*(1.0 - 4.0*t) + (4.0*s + 2.0)*t + s + 1.0;}

283 static Real H(Real t, Real s, Real u) {
284 Real t1{t*t};
285 Real t2{t1*t1};
286 Real t8{s*s};
287 Real t17{s*t8};
288 Real t23{2.0*u};
289 return
290 ((8.0*u + 6.0*t2 + 4.0)*s -
291 (6.0*t8 + 4.0*(s + u + 1.0))*t1 +
292 2.0*t8 - 4.0*t17 + t23 + 2.0)*t +
293 t1*(t8 + s + u + 1.0) +
294 (1.0 - 3.0*t2 + t23)*s +
295 u - t17 + 1.0;
296 }

306 static Real J(Real t, Real s, Real u, Real v) {
307 Real t1{s*s};
308 Real t2{t1*t1};
309 Real t17{t*t};
310 Real t22{u*u};
311 Real t32{t17*t17};
312 Real t34{t*t32};
313 Real t37{t*t17};
314 Real t46{1.0 + v};
315 Real t65{u + 1.0 + v};
316 Real t76{(-2.0*t22 + u + 4.0*v + 2.0)*u};
317 return
318 (-1.0 + 2.0*t)*(2.0 + 5.0*t)*u*t*t2 +
319 (4.0*t + 1.0)*u*s*t2 +
320 (u*t22 - 2.0*u*v - u - v - 1.0)*(4.0*t17 + 3.0*t + 1.0)*( - 1.0 + t) -
321 8.0*(t22*(t17/2.0 - 1.0/4.0) + u*(t17*t32 - 5.0/8.0*t34 - 3.0/4.0*t32 +
322 3.0/8.0*t37 + 3.0/4.0*t17 - t/8.0 - 1.0/4.0) + 3.0/4.0*t46*(t + 1.0/2.0)*(t - 2.0/3.0))*t*t1 +
323 4.0*(t22*( - 3.0/2.0*t - 1.0/4.0) + u*(t34 - t32 - 3.0/2.0*t37 + t17/4.0 - t - 1.0/4.0) -
324 t46*(t + 1.0/4.0))*s*t1 + (1.0 + v + t65*t17 - 4.0*t65*t37 - 3.0*t65*t32 + 6.0*t65*t34 +
325 t76 + 4.0*(1.0 + v + t76)*t)*s;
326 }
