544 #define CMD "Optimist::Solver::rootfind(...): "

600 template <Integer FunInDim, Integer FunOutDim, typename DerivedFunction>
601 bool solve(Function<Real, FunInDim, FunOutDim, DerivedFunction> const & function, const InputType & x_ini,
602 InputType & x_sol, bool is_optimization)
603 {
604 #define CMD "Optimist::Solver::solve(...): "
605
606 using FunctionType = Function<Real, FunInDim, FunOutDim, DerivedFunction>;
607
608 static_assert(SolInDim == FunInDim,
609 CMD "solver input dimension must be equal to the function input dimension.");
610 static_assert(SolOutDim == FunOutDim || SolOutDim == 1,
611 CMD "solver output dimension must be equal to the function output dimension or 1.");
612
613 // Lambda generators for function and derivatives
614 auto function_wrapper = [&function, is_optimization](const InputType & x, OutputType & out)
615 {
616 typename FunctionType::OutputType f;
617 function.evaluate(x, f);
618 if (is_optimization) {
619 if constexpr (FunOutDim == 1) {out = 0.5*f*f;}
620 else if constexpr (SolOutDim != FunOutDim) {out = 0.5*f.squaredNorm();}
621 else {OPTIMIST_ERROR(CMD "optimization problem with inconsistent output in function.");}
622 } else {
623 if constexpr (SolOutDim == FunOutDim) {
624 out = f;
625 }
626 else {OPTIMIST_ERROR(CMD "root-finding problem with inconsistent output in function.");}
627 }
628 };
629
630 auto first_derivative_wrapper = [&function, is_optimization, this](const InputType & x, FirstDerivativeType & out)
631 {
632 typename FunctionType::FirstDerivativeType J; function.first_derivative(x, J);
633
634 if (is_optimization) {
635 typename FunctionType::OutputType f; function.evaluate(x, f);
636 this->m_function_evaluations++;
637 if constexpr (FunInDim == 1 && FunOutDim == 1) {out = J*f;}
638 else if constexpr (SolOutDim != FunOutDim) {out = J.transpose()*f;}
639 else {OPTIMIST_ERROR(CMD "optimization problem inconsistent output in first derivative.");}
640 } else {
641 if constexpr (SolOutDim == FunOutDim) {
642 out = J;
643 }
644 else {OPTIMIST_ERROR(CMD "root-finding problem with inconsistent output in first derivative.");}
645 }
646 };
647
648 auto second_derivative_wrapper = [&function, is_optimization, this](const InputType & x, SecondDerivativeType & out)
649 {
650 typename FunctionType::SecondDerivativeType H; function.second_derivative(x, H);
651
652 if (is_optimization) {
653 typename FunctionType::OutputType f; function.evaluate(x, f);
654 typename FunctionType::FirstDerivativeType J; function.first_derivative(x, J);
655 this->m_function_evaluations++; this->m_first_derivative_evaluations++;
656 if constexpr (FunInDim == 1 && FunOutDim == 1) {out = J*J + f*H;}
657 else if constexpr (SolOutDim != FunOutDim) {
658 out = J.transpose()*J;
659 for (Integer i = 0; i < static_cast<Integer>(H.size()); ++i) {out += f(i)*H[i];}
660 }
661 else {OPTIMIST_ERROR(CMD "optimization problem with inconsistent output in second derivative.");}
662 } else {
663 if constexpr (SolOutDim == FunOutDim) {out = H;}
664 else {OPTIMIST_ERROR(CMD "root-finding problem with inconsistent output in second derivative.");}
665 }
666 };
667
668 // Select solver method based on derivative requirements
669 if constexpr (DerivedSolver::requires_function && !DerivedSolver::requires_first_derivative &&
670 !DerivedSolver::requires_second_derivative) {
671 return static_cast<DerivedSolver *>(this)->solve(function_wrapper, x_ini, x_sol);
672 } else if constexpr (DerivedSolver::requires_function && DerivedSolver::requires_first_derivative &&
673 !DerivedSolver::requires_second_derivative) {
674 return static_cast<DerivedSolver *>(this)->solve(function_wrapper, first_derivative_wrapper,
675 x_ini, x_sol);
676 } else if constexpr (DerivedSolver::requires_function && DerivedSolver::requires_first_derivative &&
677 DerivedSolver::requires_second_derivative) {
678 return static_cast<DerivedSolver *>(this)->solve(function_wrapper, first_derivative_wrapper,
679 second_derivative_wrapper, x_ini, x_sol);
680 } else {
681 OPTIMIST_ERROR(CMD "no matching function signature found for the solver.");
682 }
683
684 #undef CMD
685 }

760 bool damp(FunctionWrapper function, InputType const & x_old, InputType const & function_old,
761 InputType const & step_old, InputType & x_new, InputType & function_new, InputType & step_new)
762 {
763 Real step_norm_old, step_norm_new, residuals_old, residuals_new, tau{1.0};
764 for (this->m_relaxations = Integer(0); this->m_relaxations < this->m_max_relaxations; ++this->m_relaxations)
765 {
766 // Update point
767 step_new = tau * step_old;
768 x_new = x_old + step_new;
769 this->evaluate_function(function, x_new, function_new);
770
771 // Check relaxation
772 if constexpr (SolInDim == 1 && SolOutDim == 1) {
773 residuals_old = std::abs(function_old);
774 residuals_new = std::abs(function_new);
775 step_norm_old = std::abs(step_old);
776 step_norm_new = std::abs(step_new);
777 } else {
778 residuals_old = function_old.norm();
779 residuals_new = function_new.norm();
780 step_norm_old = step_old.norm();
781 step_norm_new = step_new.norm();
782 }
783 if (residuals_new < residuals_old || step_norm_new < (Real(1.0)-tau/Real(2.0))*step_norm_old) {
784 return true;
785 } else {
786 tau *= this->m_alpha;
787 }
788 }
789 return false;
790 }
