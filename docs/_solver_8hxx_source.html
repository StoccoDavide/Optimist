509 bool damp(Function function, InputType const & x_old, InputType const & function_old, InputType
510 const & step_old, InputType & x_new, InputType & function_new, InputType & step_new)
511 {
512 Real step_norm_old, step_norm_new, residuals_old, residuals_new, tau{1.0};
513 for (this->m_relaxations = Integer(0); this->m_relaxations < this->m_max_relaxations; ++this->m_relaxations)
514 {
515 // Update point
516 step_new = tau * step_old;
517 x_new = x_old + step_new;
518 this->evaluate_function(function, x_new, function_new);
519
520 // Check relaxation
521 if constexpr (InputDim == 1 && OutputDim == 1) {
522 residuals_old = std::abs(function_old);
523 residuals_new = std::abs(function_new);
524 step_norm_old = std::abs(step_old);
525 step_norm_new = std::abs(step_new);
526 } else {
527 residuals_old = function_old.norm();
528 residuals_new = function_new.norm();
529 step_norm_old = step_old.norm();
530 step_norm_new = step_new.norm();
531 }
532 if (residuals_new < residuals_old || step_norm_new < (Real(1.0)-tau/Real(2.0))*step_norm_old) {
533 return true;
534 } else {
535 tau *= this->m_alpha;
536 }
537 }
538 return false;
539 }
