585 protected:
596 template <Integer FunInDim, Integer FunOutDim, typename DerivedFunction>
597 bool solve(Function<FunInDim, FunOutDim, DerivedFunction> const & function, const InputType & x_ini,
598 InputType & x_sol, bool is_optimization)
599 {
600 #define CMD "Optimist::Solver::solve(...): "
601
602 using FunctionType = Function<FunInDim, FunOutDim, DerivedFunction>;
603
604 static_assert(SolInDim == FunInDim,
605 CMD "solver input dimension must be equal to the function input dimension.");
606 static_assert(SolOutDim == FunOutDim || SolOutDim == 1,
607 CMD "solver output dimension must be equal to the function output dimension or 1.");
608
609 // Lambda generators for function and derivatives
610 auto function_wrapper = [&function, is_optimization](const InputType & x, OutputType & out)
611 {
612 typename FunctionType::OutputType f;
613 function.evaluate(x, f);
614 if (is_optimization) {
615 if constexpr (FunOutDim == 1) {out = 0.5*f*f;}
616 else if constexpr (SolOutDim != FunOutDim) {out = 0.5*f.squaredNorm();}
617 else {OPTIMIST_ERROR(CMD "optimization problem with inconsistent output in function.");}
618 } else {
619 if constexpr (SolOutDim == FunOutDim) {
620 out = f;
621 }
622 else {OPTIMIST_ERROR(CMD "root-finding problem with inconsistent output in function.");}
623 }
624 };
625
626 auto first_derivative_wrapper = [&function, is_optimization, this](const InputType & x, FirstDerivativeType & out)
627 {
628 typename FunctionType::FirstDerivativeType J; function.first_derivative(x, J);
629
630 if (is_optimization) {
631 typename FunctionType::OutputType f; function.evaluate(x, f);
632 this->m_function_evaluations++;
633 if constexpr (FunInDim == 1 && FunOutDim == 1) {out = J*f;}
634 else if constexpr (SolOutDim != FunOutDim) {out = J.transpose()*f;}
635 else {OPTIMIST_ERROR(CMD "optimization problem inconsistent output in first derivative.");}
636 } else {
637 if constexpr (SolOutDim == FunOutDim) {
638 out = J;
639 }
640 else {OPTIMIST_ERROR(CMD "root-finding problem with inconsistent output in first derivative.");}
641 }
642 };
643
644 auto second_derivative_wrapper = [&function, is_optimization, this](const InputType & x, SecondDerivativeType & out)
645 {
646 typename FunctionType::SecondDerivativeType H; function.second_derivative(x, H);
647
648 if (is_optimization) {
649 typename FunctionType::OutputType f; function.evaluate(x, f);
650 typename FunctionType::FirstDerivativeType J; function.first_derivative(x, J);
651 this->m_function_evaluations++; this->m_first_derivative_evaluations++;
652 if constexpr (FunInDim == 1 && FunOutDim == 1) {out = J*J + f*H;}
653 else if constexpr (SolOutDim != FunOutDim) {
654 out = J.transpose()*J;
655 for (Integer i = 0; i < static_cast<Integer>(H.size()); ++i) {out += f(i)*H[i];}
656 }
657 else {OPTIMIST_ERROR(CMD "optimization problem with inconsistent output in second derivative.");}
658 } else {
659 if constexpr (SolOutDim == FunOutDim) {out = H;}
660 else {OPTIMIST_ERROR(CMD "root-finding problem with inconsistent output in second derivative.");}
661 }
662 };
663
664 // Select solver method based on derivative requirements
665 if constexpr (DerivedSolver::requires_function && !DerivedSolver::requires_first_derivative &&
666 !DerivedSolver::requires_second_derivative) {
667 return static_cast<DerivedSolver *>(this)->solve(function_wrapper, x_ini, x_sol);
668 } else if constexpr (DerivedSolver::requires_function && DerivedSolver::requires_first_derivative &&
669 !DerivedSolver::requires_second_derivative) {
670 return static_cast<DerivedSolver *>(this)->solve(function_wrapper, first_derivative_wrapper,
671 x_ini, x_sol);
672 } else if constexpr (DerivedSolver::requires_function && DerivedSolver::requires_first_derivative &&
673 DerivedSolver::requires_second_derivative) {
674 return static_cast<DerivedSolver *>(this)->solve(function_wrapper, first_derivative_wrapper,
675 second_derivative_wrapper, x_ini, x_sol);
676 } else {
677 OPTIMIST_ERROR(CMD "no matching function signature found for the solver.");
678 }
679
680 #undef CMD
681 }

756 bool damp(FunctionWrapper function, InputType const & x_old, InputType const & function_old,
757 InputType const & step_old, InputType & x_new, InputType & function_new, InputType & step_new)
758 {
759 Real step_norm_old, step_norm_new, residuals_old, residuals_new, tau{1.0};
760 for (this->m_relaxations = Integer(0); this->m_relaxations < this->m_max_relaxations; ++this->m_relaxations)
761 {
762 // Update point
763 step_new = tau * step_old;
764 x_new = x_old + step_new;
765 this->evaluate_function(function, x_new, function_new);
766
767 // Check relaxation
768 if constexpr (SolInDim == 1 && SolOutDim == 1) {
769 residuals_old = std::abs(function_old);
770 residuals_new = std::abs(function_new);
771 step_norm_old = std::abs(step_old);
772 step_norm_new = std::abs(step_new);
773 } else {
774 residuals_old = function_old.norm();
775 residuals_new = function_new.norm();
776 step_norm_old = step_old.norm();
777 step_norm_new = step_new.norm();
778 }
779 if (residuals_new < residuals_old || step_norm_new < (Real(1.0)-tau/Real(2.0))*step_norm_old) {
780 return true;
781 } else {
782 tau *= this->m_alpha;
783 }
784 }
785 return false;
786 }
