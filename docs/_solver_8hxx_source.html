512 template <Integer FunInDim, Integer FunOutDim, typename DerivedFunction>
513 bool solve(Function<FunInDim, FunOutDim, DerivedFunction> function, const InputType& x_ini,
514 InputType& x_sol, bool is_optimization)
515 {
516 #define CMD "Optimist::Solver::solve(...): "
517
518 using FunctionType = Function<FunInDim, FunOutDim, DerivedFunction>;
519
520 static_assert(SolInDim == FunInDim,
521 CMD "solver input dimension must be equal to the function input dimension.");
522 static_assert(SolOutDim == FunOutDim || SolOutDim == 1,
523 CMD "solver output dimension must be equal to the function output dimension or 1.");
524
525 // Lambda generators for function and derivatives
526 auto function_wrapper = [&function, is_optimization](const InputType & x, OutputType & out)
527 {
528 typename FunctionType::OutputType f;
529 function.evaluate(x, f);
530 if (is_optimization) {
531 if constexpr (FunOutDim == 1) {out = 0.5*f*f;}
532 else if constexpr (SolOutDim != FunOutDim) {out = 0.5*f.squaredNorm();}
533 else {OPTIMIST_ERROR(CMD "optimization problem with inconsistent output in function.");}
534 } else {
535 if constexpr (SolOutDim == FunOutDim) {out = f;}
536 else {OPTIMIST_ERROR(CMD "root-finding problem with inconsistent output in function.");}
537 }
538 };
539
540 auto first_derivative_wrapper = [&function, is_optimization, this](const InputType & x, FirstDerivativeType & out)
541 {
542 typename FunctionType::OutputType f; function.evaluate(x, f);
543 typename FunctionType::FirstDerivativeType J; function.first_derivative(x, J);
544 this->m_function_evaluations++;
545
546 if (is_optimization) {
547 if constexpr (FunOutDim == 1) {out = J*f;}
548 else if constexpr (SolOutDim != FunOutDim) {out = J.transpose()*f;}
549 else {OPTIMIST_ERROR(CMD "optimization problem inconsistent output in first derivative.");}
550 } else {
551 if constexpr (SolOutDim == FunOutDim) {out = J;}
552 else {OPTIMIST_ERROR(CMD "root-finding problem with inconsistent output in first derivative.");}
553 }
554 };
555
556 auto second_derivative_wrapper = [&function, is_optimization, this](const InputType & x, SecondDerivativeType & out)
557 {
558 typename FunctionType::OutputType f; function.evaluate(x, f);
559 typename FunctionType::FirstDerivativeType J; function.first_derivative(x, J);
560 typename FunctionType::SecondDerivativeType H; function.second_derivative(x, H);
561 this->m_function_evaluations++; this->m_first_derivative_evaluations++;
562
563 if (is_optimization) {
564 if constexpr (FunOutDim == 1) {out = J*J + f*H;}
565 else if constexpr (SolOutDim != FunOutDim) {
566 out = J.transpose()*J;
567 for (Integer i = 0; i < static_cast<Integer>(H.size()); ++i) {out += f(i)*H[i];}
568 }
569 else {OPTIMIST_ERROR(CMD "optimization problem with inconsistent output in second derivative.");}
570 } else {
571 if constexpr (SolOutDim == FunOutDim) {out = H;}
572 else {OPTIMIST_ERROR(CMD "root-finding problem with inconsistent output in second derivative.");}
573 }
574 };
575
576 // Select solver method based on derivative requirements
577 if constexpr (DerivedSolver::requires_function && !DerivedSolver::requires_first_derivative &&
578 !DerivedSolver::requires_second_derivative) {
579 return static_cast<DerivedSolver *>(this)->solve(function_wrapper, x_ini, x_sol);
580 } else if constexpr (DerivedSolver::requires_function && DerivedSolver::requires_first_derivative &&
581 !DerivedSolver::requires_second_derivative) {
582 return static_cast<DerivedSolver *>(this)->solve(function_wrapper, first_derivative_wrapper,
583 x_ini, x_sol);
584 } else if constexpr (DerivedSolver::requires_function && DerivedSolver::requires_first_derivative &&
585 DerivedSolver::requires_second_derivative) {
586 return static_cast<DerivedSolver *>(this)->solve(function_wrapper, first_derivative_wrapper,
587 second_derivative_wrapper, x_ini, x_sol);
588 } else {
589 OPTIMIST_ERROR(CMD "no matching function signature found for the solver.");
590 }
591
592 #undef CMD
593 }

668 bool damp(FunctionWrapper function, InputType const & x_old, InputType const & function_old,
669 InputType const & step_old, InputType & x_new, InputType & function_new, InputType & step_new)
670 {
671 Real step_norm_old, step_norm_new, residuals_old, residuals_new, tau{1.0};
672 for (this->m_relaxations = Integer(0); this->m_relaxations < this->m_max_relaxations; ++this->m_relaxations)
673 {
674 // Update point
675 step_new = tau * step_old;
676 x_new = x_old + step_new;
677 this->evaluate_function(function, x_new, function_new);
678
679 // Check relaxation
680 if constexpr (SolInDim == 1 && SolOutDim == 1) {
681 residuals_old = std::abs(function_old);
682 residuals_new = std::abs(function_new);
683 step_norm_old = std::abs(step_old);
684 step_norm_new = std::abs(step_new);
685 } else {
686 residuals_old = function_old.norm();
687 residuals_new = function_new.norm();
688 step_norm_old = step_old.norm();
689 step_norm_new = step_new.norm();
690 }
691 if (residuals_new < residuals_old || step_norm_new < (Real(1.0)-tau/Real(2.0))*step_norm_old) {
692 return true;
693 } else {
694 tau *= this->m_alpha;
695 }
696 }
697 return false;
698 }
