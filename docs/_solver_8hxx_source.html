501 bool damp(InputType const & x_old, InputType const & function_old, InputType const & step_old,
502 InputType & x_new, InputType & function_new, InputType & step_new)
503 {
504 Real step_norm_old, step_norm_new, residuals_old, residuals_new, tau{1.0};
505 for (this->m_relaxations = Integer(0); this->m_relaxations < this->m_max_relaxations; ++this->m_relaxations)
506 {
507 // Update point
508 step_new = tau * step_old;
509 x_new = x_old + step_new;
510 this->evaluate_function(x_new, function_new);
511
512 // Check relaxation
513 if constexpr (InputDim == 1 && OutputDim == 1) {
514 residuals_old = std::abs(function_old);
515 residuals_new = std::abs(function_new);
516 step_norm_old = std::abs(step_old);
517 step_norm_new = std::abs(step_new);
518 } else {
519 residuals_old = function_old.norm();
520 residuals_new = function_new.norm();
521 step_norm_old = step_old.norm();
522 step_norm_new = step_new.norm();
523 }
524 if (residuals_new < residuals_old || step_norm_new < (Real(1.0)-tau/Real(2.0))*step_norm_old) {
525 return true;
526 } else {
527 tau *= this->m_alpha;
528 }
529 }
530 return false;
