488 bool damp(InputType const & x_old, InputType const & function_old, InputType const & step_old,
489 InputType & x_new, InputType & function_new, InputType & step_new)
490 {
491 Real step_norm_old, step_norm_new, residuals_old, residuals_new, tau{1.0};
492 for (this->m_relaxations = Integer(0); this->m_relaxations < this->m_max_relaxations; ++this->m_relaxations)
493 {
494 // Update point
495 step_new = tau * step_old;
496 x_new = x_old + step_new;
497 this->evaluate_function(x_new, function_new);
498
499 // Check relaxation
500 if constexpr (InputDim == 1 && OutputDim == 1) {
501 residuals_old = std::abs(function_old);
502 residuals_new = std::abs(function_new);
503 step_norm_old = std::abs(step_old);
504 step_norm_new = std::abs(step_new);
505 } else {
506 residuals_old = function_old.norm();
507 residuals_new = function_new.norm();
508 step_norm_old = step_old.norm();
509 step_norm_new = step_new.norm();
510 }
511 if (residuals_new < residuals_old || step_norm_new < (Real(1.0)-tau/Real(2.0))*step_norm_old) {
512 return true;
513 } else {
514 tau *= this->m_alpha;
515 }
516 }
517 return false;
518 }
