88 bool bracketing(FunctionWrapper function) {
89
90 {
91 Real tolerance{0.7*this->m_tolerance_bracketing};
92 Real hba{(this->m_b - this->m_a)/2.0};
93 if (hba <= tolerance) {this->m_c = this->m_a + hba;}
94 else if (this->m_c <= this->m_a + tolerance) {this->m_c = this->m_a + tolerance;}
95 else if (this->m_c >= this->m_b - tolerance) {this->m_c = this->m_b - tolerance;}
96 }
97
98 // If f(c) = 0 set a = c and return true
99 this->evaluate_function(function, this->m_c, this->m_fc);
100 if (this->m_fc == 0) {
101 this->m_a = this->m_c; this->m_fa = 0.0;
102 this->m_d = 0.0; this->m_fd = 0.0;
103 return true;
104 } else {
105 // If f(c) is not zero, then determine the new enclosing interval
106 if (this->m_fa*this->m_fc < 0) {
107 // D <- B <- C
108 this->m_d = this->m_b; this->m_fd = this->m_fb;
109 this->m_b = this->m_c; this->m_fb = this->m_fc;
110 } else {
111 // D <- A <- C
112 this->m_d = this->m_a; this->m_fd = this->m_fa;
113 this->m_a = this->m_c; this->m_fa = this->m_fc;
114 }
115 return false;
116 }

124 Real cubic_interpolation() {
125
126 // Compute the coefficients for the inverse cubic interpolation
127 Real d_1{this->m_b - this->m_a};
128 Real d_2{this->m_d - this->m_a};
129 Real d_3{this->m_e - this->m_a};
130
131 Real dd_0{d_1/(this->m_fb - this->m_fa)};
132 Real dd_1{(d_1-d_2)/(this->m_fb - this->m_fd)};
133 Real dd_2{(d_2-d_3)/(this->m_fd - this->m_fe)};
134
135 Real ddd_0 {(dd_0 - dd_1)/(this->m_fa - this->m_fd)};
136 Real ddd_1 {(dd_1 - dd_2)/(this->m_fb - this->m_fe)};
137
138 Real dddd_0{(ddd_0-ddd_1)/(this->m_fa-this->m_fe)};
139
140 // Compute the approximate root
141 Real root{this->m_a - this->m_fa*(dd_0 - this->m_fb*(ddd_0 - this->m_fd*dddd_0))};
142 Real tolerance{0.7*this->m_tolerance_bracketing};
143 if (root <= this->m_a + tolerance || root >= this->m_b - tolerance) {
144 root = (this->m_a + this->m_b)/2.0;
145 }
146 return root;
147 }

156 Real newton_quadratic(Integer n) {
157
158 // Compute the coefficients for the quadratic interpolation
159 Real a_0{this->m_fa};
160 Real a_1{(this->m_fb - this->m_fa)/(this->m_b - this->m_a)};
161 Real a_2{((this->m_fd - this->m_fb)/(this->m_d - this->m_b)-a_1)/(this->m_d - this->m_a)};
162
163 // Safeguard to avoid overflow
164 if (a_2 == 0) {return this->m_a - a_0/a_1;}
165
166 // Determine the starting point of newton steps
167 Real c{a_2*this->m_fa > 0.0 ? this->m_a : this->m_b};
168
169 // Start the safeguarded newton steps
170 bool is_nonzero{true};
171 Real pc{0.0}, pdc{0.0};
172 for (Integer i{0}; i < n && is_nonzero ; ++i ) {
173 pc = a_0 + (a_1 + a_2*(c - this->m_b))*(c - this->m_a);
174 pdc = a_1 + a_2*((2.0 * c) - (this->m_a + this->m_b));
175 is_nonzero = pdc != 0;
176 if (is_nonzero) {c -= pc/pdc;}
177 }
178
179 if (is_nonzero) {return c;}
180 else {return this->m_a - a_0/a_1;}
181 }

196 Real find_root_impl(FunctionWrapper function) {
197
198 Real tolerance_step{this->m_tolerance_bracketing};
199 Real tolerance_function{this->m_tolerance_bracketing};
200
201 // Check for trivial solution
202 this->m_converged = this->m_fa == 0; if (this->m_converged) {return this->m_a;}
203 this->m_converged = this->m_fb == 0; if (this->m_converged) {return this->m_b;}
204
205 // Initialize to dumb values
206 this->m_e = QUIET_NAN;
207 this->m_fe = QUIET_NAN;
208
209 // While f(left) or f(right) are infinite perform bisection
210 while (!(std::isfinite(this->m_fa) && std::isfinite(this->m_fb))) {
211 ++this->m_iterations;
212 this->m_c = (this->m_a + this->m_b)/2.0;
213 this->evaluate_function(function, this->m_c, this->m_fc);
214 this->m_converged = this->m_fc == 0;
215 if (this->m_converged) {return this->m_c;}
216 if (this->m_fa*this->m_fc < 0.0) {
217 // -> [a, c]
218 this->m_b = this->m_c; this->m_fb = this->m_fc;
219 } else {
220 // -> [c, b]
221 this->m_a = this->m_c; this->m_fa = this->m_fc;
222 }
223
224 // Check for convergence
225 Real abs_fa{std::abs(this->m_fa)};
226 Real abs_fb{std::abs(this->m_fb)};
227 this->m_converged = (this->m_b - this->m_a) < tolerance_step ||
228 abs_fa < tolerance_function ||abs_fb < tolerance_function;
229 if (this->m_converged) {return abs_fb < abs_fa ? this->m_b : this->m_a;}
230 }
231
232 {
233 // Impede the step c to be too close to the bounds a or b.
234 Real b_a{this->m_b - this->m_a};
235 Real r{b_a/(this->m_fb - this->m_fa)};
236 this->m_c = std::abs(this->m_fb) < std::abs(this->m_fa) ?
237 this->m_b + this->m_fb*r : this->m_a - this->m_fa*r;
238 Real delta{this->m_interval_shink*b_a}, a_min, b_max;
239 if (this->m_c < (a_min = this->m_a + delta)) {this->m_c = a_min;}
240 else if (this->m_c > (b_max = this->m_b - delta)) {this->m_c = b_max;}
241 }
242
243 // Call "bracketing" to get a shrinked enclosing interval and to update the termination criterion
244 this->m_converged = this->bracketing(function);
245 if (this->m_converged ) {return this->m_a;}
246 this->m_converged = false;
247
248 // The enclosing interval is recorded as [a0, b0] before executing the iteration steps
249 while (++this->m_iterations < this->m_max_iterations) {
250 ++this->m_iterations;
251 Real a0_b0{this->m_b - this->m_a};
252
253 // Check the termination criterion
254 {
255 Real abs_fa {std::abs(this->m_fa)};
256 Real abs_fb {std::abs(this->m_fb)};
257 this->m_converged = abs_fa < tolerance_function || abs_fb < tolerance_function;
258 if (this->m_converged) {return abs_fa < abs_fb ? this->m_a : this->m_b;}
259 }
260
261 // Starting with the second iteration: newton quadratic or cubic interpolation
262 bool do_newton_quadratic{false};
263 if (!std::isfinite(this->m_fe)) {
264 do_newton_quadratic = true;
265 } else if (!this->all_different( this->m_fa, this->m_fb, this->m_fd, this->m_fe)) {
266 do_newton_quadratic = true;
267 } else {
268 this->m_c = this->cubic_interpolation();
269 do_newton_quadratic = (this->m_c-this->m_a)*(this->m_c-this->m_b) >= 0.0;
270 }
271 if (do_newton_quadratic) {this->m_c = this->newton_quadratic(2);}
272
273 this->m_e = this->m_d;
274 this->m_fe = this->m_fd;
275
276 // Call "bracketing" to get a shrinked enclosing interval and to update the termination criterion
277 this->m_converged = this->bracketing(function) || (this->m_b-this->m_a) <= this->m_tolerance;
278 if (this->m_converged) {
279 return std::abs(this->m_fa) < std::abs(this->m_fb) ? this->m_a : this->m_b;
280 }
281 if (!this->all_different(this->m_fa, this->m_fb, this->m_fd, this->m_fe)) {
282 do_newton_quadratic = true;
283 } else {
284 this->m_c = this->cubic_interpolation();
285 do_newton_quadratic = (this->m_c - this->m_a)*(this->m_c - this->m_b) >= 0.0;
286 }
287 if (do_newton_quadratic) {this->m_c = this->newton_quadratic(3);}
288
289 // Call "bracketing" to get a shrinked enclosing interval and to update the termination criterion
290 {
291 this->m_converged = this->bracketing(function) || (this->m_b-this->m_a) <= this->m_tolerance;
292 Real abs_fa{std::abs(this->m_fa)};
293 Real abs_fb{std::abs(this->m_fb)};
294 if (this->m_converged) {return abs_fa < abs_fb ? this->m_a : this->m_b;}
295
296 this->m_e = this->m_d;
297 this->m_fe = this->m_fd;
298
299 // Takes the double-size secant step
300 Real u, fu;
301 if (abs_fa < abs_fb) {u = this->m_a; fu = this->m_fa;}
302 else {u = this->m_b; fu = this->m_fb;}
303 Real hba{(this->m_b - this->m_a)/2.0};
304 this->m_c = u - 4.0*(fu/(this->m_fb - this->m_fa))*hba;
305 if (std::abs(this->m_c - u) > hba) {this->m_c = this->m_a + hba;}
306 }
307
308 // Call "bracketing" to get a shrinked enclosing interval and to update the termination criterion
309 this->m_converged = this->bracketing(function) || (this->m_b-this->m_a) <= this->m_tolerance_bracketing;
310 if (this->m_converged) {
311 return std::abs(this->m_fa) < std::abs(this->m_fb) ? this->m_a : this->m_b;
312 }
313
314 // Determine whether an additional bisection step is needed
315 if ((this->m_b-this->m_a) < this->m_mu*a0_b0) {continue;}
316
317 this->m_e = this->m_d;
318 this->m_fe = this->m_fd;
319
320 // Call "bracketing" to get a shrinked enclosing interval and to update the termination criterion
321 {
322 Real b_a{this->m_b-this->m_a};
323 this->m_c = this->m_a + b_a/2.0;
324 this->m_converged = this->bracketing(function) || b_a <= this->m_tolerance_bracketing;
325 }
326 }
327 // Return the approximate root
328 return std::abs(this->m_fa) < std::abs(this->m_fb) ? this->m_a : this->m_b;
329 }
