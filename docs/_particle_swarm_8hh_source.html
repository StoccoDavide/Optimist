308 void randomize_particles(Particles & particles) const
309 {
310 for(Integer i{0}; i < particles.size(); ++i) {
311 for(Integer j{0}; j < particles.rows(); ++j) {
312 Scalar lb{this->m_lower_bound(j)};
313 Scalar ub{this->m_upper_bound(j)};
314 particles(j, i) = lb + (this->m_dice() * (ub - lb));
315 }
316 }
317 }
318
323 void randomize_velocities(Particles & velocities) const
324 {
325 for (Integer i{0}; i < velocities.cols(); ++i) {
326 for (Integer j{0}; j < velocities.rows(); ++j) {
327 Scalar lb{this->m_lower_bound(j)};
328 Scalar ub{this->m_upper_bound(j)};
329 Scalar diff{ub - lb};
330 Scalar vel{-diff + (this->m_dice() * 2.0 * diff)};
331 velocities(j, i) = std::min(m_max_velocity, std::max(-m_max_velocity, vel));
332 }
333 }
334 }

336 void evaluate_objective(Particles const & particles, Vector & fvals)
337 {
338 for(Integer i{0}; i < particles.size(); ++i) {fvals(i) = objective_(particles.col(i));}

341 void maintainBounds(Matrix &particles) const
342 {
343 for(Integer i{0}; i < particles.size(); ++i)
344 {
345 for(Integer j{0}; j < particles.rows(); ++j)
346 {
347 Scalar minval{bounds(0, j)};
348 Scalar maxval{bounds(1, j)};
349 particles[j](i) = std::min(maxval, std::max(minval, particles[j](i)));
350 }

354 void calculateVelocities(const Matrix &particles,
355 const Matrix &best_particles,
356 Integer const gbest,
357 Integer const iteration,
358 Matrix &velocities)
359 {
360 assert(velocities.rows() == particles.rows());
361 assert(velocities.cols() == particles.size());
362 assert(velocities.rows() == best_particles.rows());
363 assert(velocities.cols() == best_particles.size());
364 assert(gbest < best_particles.size());
365
366 Scalar weight = m_weight_strategy(iteration, this->m_max_iterations);
367
368 for(Integer i{0}; i < velocities.cols(); ++i)
369 {
370 for(Integer j{0}; j < velocities.rows(); ++j)
371 {
372 Scalar velp = this->m_dice() * (best_particles(j, i) - particles(j, i));
373 Scalar velg = this->m_dice() * (best_particles(j, gbest) - particles(j, i));
374 Scalar vel = weight * velocities(j, i) + m_phi_p * velp + m_phi_g * velg;
375
376 if(m_max_velocity > 0)
377 vel = std::min(m_max_velocity, std::max(-m_max_velocity, vel));
378
379 velocities(j, i) = vel;
380 }
381 }
382 }

384 Result _minimize(const Matrix &bounds, Matrix &particles)
385 {
386 Matrix velocities(particles.rows(), particles.size());
387
388 Vector fvals(particles.size());
389
390 Matrix best_particles = particles;
391 Vector bestFvals(particles.size());
392
393 Matrix prevParticles(particles.rows(), particles.size());
394 Vector prevFvals(particles.size());
395
396 Vector diff(particles.rows());
397
398 Index gbest = 0;
399
400 // initialize velocities randomly
401 randomize_velocities(bounds, velocities);
402
403 // evaluate objective function for the initial particles
404 evaluate_objective(particles, fvals);
405 bestFvals = fvals;
406 bestFvals.minCoeff(&gbest);
407
408 // init stop conditions
409 Index iterations = 0;
410 Scalar fchange = feps_ + 1;
411 Scalar xchange = xeps_ + 1;
412
413 while((this->m_iterations < this->m_max_iterations) &&
414 fchange > feps_ && xchange > xeps_)
415 {
416 // calculate new velocities
417 calculateVelocities(particles, best_particles, gbest, iterations, velocities);
418
419 // move particles by velocity and stay within bounds
420 particles += velocities;
421 maintainBounds(bounds, particles);
422
423 // evaluate objective for moved particles
424 evaluate_objective(particles, fvals);
425
426 prevParticles = best_particles;
427 prevFvals = bestFvals;
428
429 for(Integer i{0}; i < fvals.size(); ++i)
430 {
431 // check if there was an improvement and update best vals
432 if(fvals(i) < bestFvals(i))
433 {
434 bestFvals(i) = fvals(i);
435 best_particles.col(i) = particles.col(i);
436 }
437 }
438 bestFvals.minCoeff(&gbest);
439
440 // calculate new diffs
441 xchange = (best_particles - prevParticles).colwise().norm().sum();
442 fchange = (bestFvals - prevFvals).array().abs().sum();
443
444 xchange /= best_particles.size();
445 fchange /= bestFvals.size();
446
447 if(this->m_verbose) {this->info(gbest);}
448
449 ++iterations;
450 }
451
452 result.iterations = iterations;
453 result.converged = fchange <= feps_ || xchange <= xeps_;
454 result.fval = bestFvals(gbest);
455 result.xval = best_particles.col(gbest);
456
457 return result;
458 }

544 Result minimize(Integer const cnt)
545 {
546 if(cnt == 0)
547 throw std::runtime_error("particle count cannot be 0");
548 if(bounds.rows() != 2)
549 throw std::runtime_error("bounds has not exactly 2 rows (min, max)");
550 for(Integer i{0}; i < bounds.cols(); ++i)
551 {
552 if(bounds(0, i) >= bounds(1, i))
553 throw std::runtime_error("bounds min is greater than max");
554 }
555
556 Matrix particles(bounds.cols(), cnt);
557 randomize_particles(bounds, particles);
558
559 return _minimize(bounds, particles);
560 }

579 Result minimize(Integer const cnt, const Vector &initGuess)
580 {
581 if(cnt == 0)

583 if(bounds.rows() != 2)
584 throw std::runtime_error("bounds has not exactly 2 rows (min, max)");
585 for(Integer i{0}; i < bounds.cols(); ++i)

588 throw std::runtime_error("bounds min is greater than max");
