115 bool solve(Vector const &x_ini, Vector &x_sol) override
116 {
117 // Setup internal variables
118 this->reset();
119
120 // Initialize variables
121 Real residuals_old, residuals_new, step_norm_old, step_norm_new, tau;
122 Vector x_old, x_new, function_old, function_new, step_old, step_new, delta_x_old, delta_x_new,
123 delta_function_old, delta_function_new;
124 Matrix jacobian_old, jacobian_new;
125
126 // Set initial iteration
127 x_old = x_ini;
128 this->evaluate_function(x_old, function_old);
129 this->evaluate_jacobian(x_old, jacobian_old);
130
131 // Algorithm iterations
132 Real tolerance_residuals{this->m_tolerance};
133 Real tolerance_step_norm{this->m_tolerance * this->m_tolerance};
134 for (this->m_iterations = Integer(1); this->m_iterations < this->m_max_iterations; ++this->m_iterations)
135 {
136
137 // Calculate step
138 step_old = -jacobian_old * function_old;
139
140 // Check convergence
141 residuals_old = function_old.norm();
142 step_norm_old = step_old.norm();
143 this->store_trace(x_old);
144 if (residuals_old < tolerance_residuals || step_norm_old < tolerance_step_norm) {
145 this->m_converged = true;
146 break;
147 }
148
149 if (this->m_damped)
150 {
151 // Relax the iteration process
152 tau = Real(1.0);
153 for (this->m_relaxations = Integer(0); this->m_relaxations < this->m_max_relaxations; ++this->m_relaxations)
154 {
155 // Update point
156 step_new = tau * step_old;
157 x_new = x_old + step_new;
158 this->evaluate_function(x_new, function_new);
159
160 // Check relaxation
161 residuals_new = function_new.norm();
162 step_norm_new = step_new.norm();
163 if (residuals_new < residuals_old || step_norm_new < (Real(1.0)-tau/Real(2.0))*step_norm_old) {
164 break;
165 } else {
166 tau *= this->m_alpha;
167 }
168 }
169 } else {
170 // Update point
171 x_new = x_old + step_old;
172 this->evaluate_function(x_new, function_new);
173 }
174
175 // Update jacobian approximation
176 delta_x_new = x_new - x_old;
177 delta_function_new = function_new - function_old;
178 this->update(
179 delta_x_old, delta_function_old, jacobian_old, // Old step data
180 delta_x_new, delta_function_new, jacobian_new // New step data
181 );
182
183 // Update internal variables
184 x_old = x_new;
185 function_old = function_new;
186 delta_x_old = delta_x_new;
187 delta_function_old = delta_function_new;
188 step_old = step_new;
189 jacobian_old = jacobian_new;
190 }
191
192 // Convergence data
193 x_sol = x_old;
194 return this->m_converged;
195 }

206 void update(
207 Vector const &delta_x_old, Vector const &delta_function_old, Matrix const &jacobian_old,
208 Vector const &delta_x_new, Vector const &delta_function_new, Matrix &jacobian_new
209 ) {
210 Vector tmp_1(jacobian_old * delta_function_new);
211 Real tmp_2{delta_function_new.squaredNorm()};
212 // Selection criteria: |(dx_new'*dx_old) / (dx_new'*J_old*dF_new)| < |(dF_new'*dF_old) / (dF_new'*dF_new)|
213 if (this->m_method == Method::COMBINED || this->m_method == Method::GOOD || this->iterations() < Integer(2) ||
214 std::abs(delta_x_new.transpose() * delta_x_old) / std::abs(delta_x_new.transpose() * tmp_1)
215 < std::abs(delta_function_new.transpose() * delta_function_old) / tmp_2) {
216 // Broyden's Good solver: J_new = J_old - (J_old*dF_new-dx_new) / (C'*dF_new)*C', where C = J_old'*dx_new;
217 Vector C_g(jacobian_old.transpose() * delta_x_new);
218 jacobian_new = jacobian_old - (tmp_1 - delta_x_new) / (C_g.transpose() * delta_function_new) * C_g.transpose();
219 } else {
220 // Broyden's Bad solver: J_new = J_old - (J_old*dF_old-dx_new) / (C'*dF_old)*C', where C = dF_old;
221 jacobian_new = jacobian_old - (tmp_1 - delta_x_new) / tmp_2 * delta_function_old.transpose();
222 }
223 }
