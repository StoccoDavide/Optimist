125 void update_impl(
126 Vector const & delta_x_old, Vector const & delta_function_old, Matrix const & jacobian_old,
127 Vector const & delta_x_new, Vector const & delta_function_new, Vector const & /*function_new*/,
128 Matrix & jacobian_new
129 ) {
130 Vector tmp_1(jacobian_old * delta_function_new);
131 Real tmp_2{delta_function_new.squaredNorm()};
132 // Selection criteria: |(dx_new'*dx_old) / (dx_new'*J_old*dF_new)| < |(dF_new'*dF_old) / (dF_new'*dF_new)|
133 if (this->m_method == Method::COMBINED || this->m_method == Method::GOOD || this->iterations() < Integer(2) ||
134 std::abs(delta_x_new.transpose() * delta_x_old) / std::abs(delta_x_new.transpose() * tmp_1)
135 < std::abs(delta_function_new.transpose() * delta_function_old) / tmp_2) {
136 // Broyden's Good solver: J_new = J_old - (J_old*dF_new-dx_new) / (C'*dF_new)*C', where C = J_old'*dx_new;
137 Vector C_g(jacobian_old.transpose() * delta_x_new);
138 jacobian_new = jacobian_old - (tmp_1 - delta_x_new) / (C_g.transpose() * delta_function_new) * C_g.transpose();
139 } else {
140 // Broyden's Bad solver: J_new = J_old - (J_old*dF_old-dx_new) / (C'*dF_old)*C', where C = dF_old;
141 jacobian_new = jacobian_old - (tmp_1 - delta_x_new) / tmp_2 * delta_function_old.transpose();
142 }
143 }
