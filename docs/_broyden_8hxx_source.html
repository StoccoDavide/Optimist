122 void update_impl(
123 Vector const & delta_x_old, Vector const & delta_function_old, Matrix const & jacobian_old,
124 Vector const & delta_x_new, Vector const & delta_function_new, Vector const & /*function_new*/,
125 Matrix & jacobian_new
126 ) {
127 Vector tmp_1(jacobian_old * delta_function_new);
128 Real tmp_2{delta_function_new.squaredNorm()};
129 // Selection criteria: |(dx_new'*dx_old) / (dx_new'*J_old*dF_new)| < |(dF_new'*dF_old) / (dF_new'*dF_new)|
130 if (this->m_method == Method::COMBINED || this->m_method == Method::GOOD || this->iterations() < Integer(2) ||
131 std::abs(delta_x_new.transpose() * delta_x_old) / std::abs(delta_x_new.transpose() * tmp_1)
132 < std::abs(delta_function_new.transpose() * delta_function_old) / tmp_2) {
133 // Broyden's Good solver: J_new = J_old - (J_old*dF_new-dx_new) / (C'*dF_new)*C', where C = J_old'*dx_new;
134 Vector C_g(jacobian_old.transpose() * delta_x_new);
135 jacobian_new = jacobian_old - (tmp_1 - delta_x_new) / (C_g.transpose() * delta_function_new) * C_g.transpose();
136 } else {
137 // Broyden's Bad solver: J_new = J_old - (J_old*dF_old-dx_new) / (C'*dF_old)*C', where C = dF_old;
138 jacobian_new = jacobian_old - (tmp_1 - delta_x_new) / tmp_2 * delta_function_old.transpose();
139 }
140 }
