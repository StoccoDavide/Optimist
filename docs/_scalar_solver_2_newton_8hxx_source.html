74 bool solve(Real const &x_ini, Real &x_sol) override
75 {
76 // Setup internal variables
77 this->reset();
78
79 // Print header
80 if (this->m_verbose) {this->header();}
81
82 // Initialize variables
83 Real residuals_old, residuals_new, step_norm_old, step_norm_new, tau;
84 Real x_old, x_new, function_old, function_new, step_old, step_new;
85 Real first_derivative;
86
87 // Set initial iteration
88 x_old = x_ini;
89 this->evaluate_function(x_old, function_old);
90 this->evaluate_first_derivative(x_old, first_derivative);
91
92 // Algorithm iterations
93 Real tolerance_residuals{this->m_tolerance};
94 Real tolerance_step_norm{this->m_tolerance * this->m_tolerance};
95 for (this->m_iterations = Integer(1); this->m_iterations < this->m_max_iterations; ++this->m_iterations)
96 {
97 // Store trace
98 this->store_trace(x_old);
99
100 // Calculate step
101 OPTIMIST_ASSERT(std::abs(first_derivative) > EPSILON_LOW,
102 "Optimist::ScalarSolver::Newton::solve(...): singular first_derivative detected.");
103 step_old = -function_old/first_derivative;
104
105 // Check convergence
106 residuals_old = std::abs(function_old);
107 step_norm_old = std::abs(step_old);
108 if (this->m_verbose) {this->info(residuals_old);}
109 if (residuals_old < tolerance_residuals || step_norm_old < tolerance_step_norm) {
110 this->m_converged = true;
111 break;
112 }
113
114 if (this->m_damped)
115 {
116 // Relax the iteration process
117 tau = Real(1.0);
118 for (this->m_relaxations = Integer(0); this->m_relaxations < this->m_max_relaxations; ++this->m_relaxations)
119 {
120 // Update point
121 step_new = tau * step_old;
122 x_new = x_old + step_new;
123 this->evaluate_function(x_new, function_new);
124
125 // Check relaxation
126 residuals_new = std::abs(function_new);
127 step_norm_new = std::abs(step_new);
128 if (residuals_new < residuals_old || step_norm_new < (Real(1.0)-tau/Real(2.0))*step_norm_old) {
129 this->evaluate_first_derivative(x_new, first_derivative);
130 break;
131 } else {
132 tau *= this->m_alpha;
133 }
134 }
135 } else {
136 // Update point
137 x_new = x_old + step_old;
138 this->evaluate_function(x_new, function_new);
139 this->evaluate_first_derivative(x_new, first_derivative);
140 }
141
142 // Update internal variables
143 x_old = x_new;
144 function_old = function_new;
145 step_old = step_new;
146 }
147
148 // Print bottom
149 if (this->m_verbose) {this->bottom();}
150
151 // Convergence data
152 x_sol = x_old;
153 return this->m_converged;
154 }
