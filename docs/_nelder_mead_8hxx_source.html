38 template <Integer N>
39 class NelderMead : public Optimizer<N, NelderMead<N>>
40 {
41 public:
42 static constexpr bool requires_function = true;
43 static constexpr bool requires_first_derivative = false;
44 static constexpr bool requires_second_derivative = false;
45
46 using Vector = typename Optimizer<N, NelderMead<N>>::Vector;
47 using Matrix = typename Optimizer<N, NelderMead<N>>::Matrix;
48 using FunctionWrapper = typename Optimizer<N, NelderMead<N>>::FunctionWrapper;
49 using Optimizer<N, NelderMead<N>>::solve;
50
51 using Simplex = std::vector<Vector>;
52
53 private:
54 Simplex m_s;
55 Vector m_c;
56 Real m_alpha{1.0};
57 Real m_gamma{2.0};
58 Real m_rho{0.5};
59 Real m_sigma{0.5};
60
61 public:
65 NelderMead() {
66 this->m_s.resize(N);
67 }
68
73 std::string name_impl() const {return "NelderMead";}
74
80 void initialize(const Vector & x_ini, Real delta)
81 {
82 this->m_s.clear();
83 this->m_s.push_back(x_ini);
84 Vector x;
85 for (Integer i{0}; i < N; ++i) {
86 x = x_ini;
87 x(i) += delta;
88 this->m_s.push_back(x);
89 }
90 this->m_c = this->centroid();
91 }
92
97 Vector centroid() const
98 {
99 return 1.0/Real(N) * std::accumulate(this->m_s.begin(), this->m_s.end(), this->m_c);
100 }
101
105 void sort(FunctionWrapper function)
106 {
107 std::sort(this->m_s.begin(), this->m_s.end(),
108 [function, this] (const Vector & a, const Vector & b) {
109 Real function_a, function_b;
110 this->evaluate_function(function, a, function_a);
111 this->evaluate_function(function, b, function_b);
112 return function_a < function_b;
113 }
114 );
115 this->m_c = this->centroid();
116 }
117
122 Vector reflection() const
123 {
124 return this->m_c + this->m_alpha * (this->m_c - this->m_s.back());
125 }
126
132 Vector expansion(const Vector & x) const
133 {
134 return this->m_c + this->m_gamma * (x - this->m_c);
135 }
136
142 Vector outward_contraction(const Vector & x) const
143 {
144 return this->m_c + this->m_rho * (x - this->m_c);
145 }
146
152 Vector inward_contraction() const
153 {
154 return this->m_c + this->m_rho * (this->m_s.back() - this->m_c);
155 }
156
160 void shrink()
161 {
162 for (Integer i{1}; i < static_cast<Integer>(this->m_s.size()); ++i) {
163 this->m_s.at(i) = this->m_s.at(1) + this->m_sigma * (this->m_s.at(i) - this->m_s.at(1));
164 }
165 }

175 bool solve_impl(FunctionWrapper function, Vector const &x_ini, Vector &x_sol)
176 {
177 // Setup internal variables
178 this->reset();
179
180 // Print header
181 if (this->m_verbose) {this->header();}
182
183 // Initialize variables
184 Vector x_r, x_c, x_e, simplex_size;
185 Real function_x_r, function_x_1, function_x_last, function_x_e, function_x_c;
186
187 // Set initial iteration
188 this->initialize(x_ini, 100.0);
189
190 // Algorithm iterations
191 for (this->m_iterations = Integer(1); this->m_iterations < this->m_max_iterations; ++this->m_iterations)
192 {
193 // Sort the simplex points based on the function values
194 this->sort(function);
195
196 // Store trace
197 this->store_trace(this->m_s.back());
198
199 // Compute the reflection point
200 x_r = this->reflection();
201 this->evaluate_function(function, x_r, function_x_r);
202 this->evaluate_function(function, this->m_s.at(1), function_x_1);
203 this->evaluate_function(function, this->m_s.back(), function_x_last);
204
205 if (function_x_1 <= function_x_r && function_x_r < function_x_last)
206 {
207 this->m_s.back() = x_r;
208 if (this->m_verbose) {this->info(function_x_1, "Reflection");}
209 }
210 else if (function_x_r < function_x_1)
211 {
212 // Expansion
213 x_e = this->expansion(x_r);
214 this->evaluate_function(function, x_e, function_x_e);
215 if (function_x_e < function_x_r) {
216 this->m_s.back() = x_e;
217 } else {
218 this->m_s.back() = x_r;
219 }
220 if (this->m_verbose) {this->info(function_x_1, "Expansion");}
221 }
222 else if (function_x_r < function_x_last)
223 {
224 // Outward contraction
225 x_c = this->outward_contraction(x_r);
226 this->evaluate_function(function, x_c, function_x_c);
227 if (function_x_c < function_x_r) {
228 this->m_s.back() = x_c;
229 } else {
230 this->shrink();
231 }
232 if (this->m_verbose) {this->info(function_x_1, "Outward contraction");}
233 }
234 else
235 {
236 // Inward contraction
237 x_c = this->inward_contraction();
238 this->evaluate_function(function, x_c, function_x_c);
239 if (function_x_c < function_x_last) {
240 this->m_s.back() = x_c;
241 } else {
242 this->shrink();
243 }
244 if (this->m_verbose) {this->info(function_x_1, "Inward contraction");}
245 }
246 simplex_size = this->m_s.back() - this->m_s.front();
247 if (simplex_size.norm() < this->m_tolerance) {
248 this->m_converged = true;
249 break;
250 }
251 }
252
253 // Print bottom
254 if (this->m_verbose) {this->bottom();}
255
256 // Convergence data
257 x_sol = this->m_s.at(0);
258 return this->m_converged;
259 }
