79 // SEARCH This method call the explore method on the first
80 // iteration and then continue to call explore until a stencil
81 // fails. In the case of a stencil failure, it tries once to go
82 // back of half a step along the search direction by setting x_center
83 // equal to the base point x_best.
84 // If the stencil fails again, it exits the while loop and stencil_failure
85 // is set to zero in order to signal that a reduction of h is necessary.
86 void
87 search() {
88 ++m_iteration_count; // augment counter
89
90 this->best_nearby();
91
92 while ( m_stencil_failure ) {
93 // reduce the scale
94 m_h *= m_rho;
95 this->best_nearby();
96 if ( m_h <= m_tolerance ) return;
97 if ( m_fun_evaluation_count >= m_max_fun_evaluation ) return;
98 }
99
100 search_dir.noalias() = x_best - x_old; // Compute search direction
101
102 // Continue exploring until stencil failure or exceed of
103 Real lambda = 1;
104 Real max_der = 0;
105 while ( m_fun_evaluation_count < m_max_fun_evaluation && lambda > 0.1 ) {
106 x_new.noalias() = x_best + lambda*search_dir;
107 Real new_f = eval_function(x_new);
108 if ( new_f < f_best-(0.25*lambda)*max_der ) {
109 Real der = (f_best-new_f)/lambda;
110 if ( der > max_der ) max_der = der;
111 x_best.noalias() = x_new;
112 f_best = new_f;
113 //lambda *= 2;
114 }
115 lambda *= 0.5;
116 }
117 }
118
119 // This method explore all points on the stencil center at
120 // x_temporary = x_center and updates the current iteration x to the current
121 // best point x_current_best. If the current best point x_current_best is worse than the
122 // base point x_best, the current iteration x will remain constant
123 // (x = x_best) and stencil failure flag stencil_failure will be set to zero.
124 void best_nearby()
125 {
126 /*
127 */
128 // Initialize
129 m_stencil_failure = true;
130
131 // ----------------------------------------------------------------------------------------
132 // Cycle on all stencil directions
133
134 for ( integer j = 0; j < N; ++j ) {
135 Real s_dirh = search_sign(j) * m_h;
136 m_p.noalias() = x_best; m_p(j) += s_dirh;
137 Real fp = eval_function( m_p );
138 if ( fp >= f_best ) {
139 m_p1.noalias() = x_best; m_p1(j) -= s_dirh; // try the opposite direction
140 Real fp1 = eval_function( m_p1 );
141 if ( fp1 < fp ) {
142 m_p.noalias() = m_p1; fp = fp1;
143 // change priority of search direction to the opposite verse
144 search_sign(j) = -search_sign(j);
145 }
146 }
147 // Update temporary and current best point before checking
148 // the remaining directions j
149 if ( fp < f_best ) {
150 x_best.noalias() = m_p; // move temporary point
151 f_best = fp; // new current best point
152 m_stencil_failure = false; // update stencil failure flag
153 }
154 }
155 }

165 bool solve_impl(FunctionWrapper function, Vector const &x_ini, Vector &x_sol)
166 {
167 // Setup internal variables
168 this->reset();
169 this->m_stencil_failure = false;
170
171 // Print header
172 if (this->m_verbose) {this->header();}
173
174 // Set initial iteration
175 Vector x_best(x_ini);
176 Real f_best;
177 this->evaluate_function(function, x_best, f_best);
178 search_sign.setOnes(); // First direction will be positive for each direction
179 this->m_h = h;
180
181 // Algorithm iterations
182 integer stagnations{0};
183 for (this->m_iterations = Integer(0); this->m_iterations < this->m_max_iterations; ++this->m_iterations)
184 {
185
186 x_old = x_best;
187 f_old = f_best;
188
189 this->search();
190 if (this->m_stencil_failure) {break;}
191
192 // Check convergence
193 if (this->m_verbose) {this->info(residuals);}
194 if (this->m_h < this->m_tolerance) {
195 this->m_converged = true;
196 break;
197 }
198
199 // Reduce the scale
200 this->m_h *= this->m_rho;
201
202 // Check for stagnation
203 if (f_old <= f_best) {
204 ++stagnations;
205 if (stagnations > this->m_max_stagnations) {break;}
206 } else {
207 stagnations = 0;
208 }
209 }
210
211 // Print bottom
212 if (this->m_verbose) {this->bottom();}
213
214 // Convergence data
215 x_sol =
216 return this->m_converged;
217 }
