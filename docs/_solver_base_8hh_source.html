186 void reset_bounds(Integer const n = InputTrait::IsDynamic ? 0 : InputTrait::Dimension)
187 {
188 #define CMD "Optimist::Solver::reset_bounds(...): "
189
190 if constexpr (InputTrait::IsScalar) {
191 this->m_lower_bound = -INFTY;
192 this->m_upper_bound = +INFTY;
193 } else if constexpr (InputTrait::IsFixed) {
194 this->m_lower_bound.setConstant(-INFTY);
195 this->m_upper_bound.setConstant(+INFTY);
196 } else if constexpr (InputTrait::IsDynamic) {
197 this->m_lower_bound.resize(n); this->m_lower_bound.setConstant(-INFTY);
198 this->m_upper_bound.resize(n); this->m_upper_bound.setConstant(+INFTY);
199 } else if constexpr (InputTrait::IsSparse) {
200 this->m_lower_bound.resize(n); this->m_lower_bound.reserve(n);
201 this->m_upper_bound.resize(n); this->m_upper_bound.reserve(n);
202 std::vector<Eigen::Triplet<Scalar>> triplets; triplets.reserve(n);
203 for (Integer i{0}; i < n; ++i) {triplets.emplace_back(i, 0, -INFTY);}
204 this->m_lower_bound.setFromTriplets(triplets.begin(), triplets.end());
205 triplets.clear(); triplets.reserve(n);
206 for (Integer i{0}; i < n; ++i) {triplets.emplace_back(i, 0, +INFTY);}
207 this->m_upper_bound.setFromTriplets(triplets.begin(), triplets.end());
208 } else {
209 OPTIMIST_ERROR(CMD "unsupported input type for bounds reset.");
210 }
211
212 #undef CMD
213 }

685 template <typename FunctionInput, typename FunctionOutput, typename DerivedFunction>
686 // TODO: make requires clauses to avoid invalid combinations
687 bool solve(FunctionBase<FunctionInput, FunctionOutput, DerivedFunction>
688 const & function, Input const & x_ini, Input & x_sol, bool is_optimization)
689 {
690 #define CMD "Optimist::Solver::solve(...): "
691
692 using FunctionType = FunctionBase<FunctionInput, FunctionOutput, DerivedFunction>;
693 using FunctionInputTrait = TypeTrait<FunctionInput>;
694 using FunctionOutputTrait = TypeTrait<FunctionOutput>;
695
696 static_assert(InputTrait::Dimension == FunctionInputTrait::Dimension,
697 CMD "solver input dimension must be equal to the function input dimension.");
698 static_assert(OutputTrait::Dimension == FunctionOutputTrait::Dimension || OutputTrait::Dimension == 1,
699 CMD "solver output dimension must be equal to the function output dimension or a scalar.");
700
701 // Lambda generators for function and derivatives
702 auto function_lambda = [&function, is_optimization] (Input const & x, Output & out) -> bool
703 {
704 bool success{false};
705 FunctionOutput f; success = function.evaluate(x, f);
706 OPTIMIST_ASSERT(success,
707 CMD "function evaluation failed during function computation.");
708
709 if (is_optimization) {
710 if constexpr (FunctionOutputTrait::Dimension == 1) {
711 out = 0.5*f*f;
712 } else if constexpr (OutputTrait::Dimension != FunctionOutputTrait::Dimension) {
713 out = 0.5*f.squaredNorm();
714 } else {
715 OPTIMIST_ERROR(CMD "optimization problem with inconsistent output in function.");
716 }
717 } else {
718 if constexpr (OutputTrait::Dimension == FunctionOutputTrait::Dimension) {
719 out = f;
720 } else {
721 OPTIMIST_ERROR(CMD "root-finding problem with inconsistent output in function.");
722 }
723 }
724 return success;
725 };
726
727 auto first_derivative_lambda = [&function, is_optimization, this] (Input const & x,
728 FirstDerivative & out) -> bool
729 {
730 bool success{false};
731 typename FunctionType::FirstDerivative J; success = function.first_derivative(x, J);
732 OPTIMIST_ASSERT(success,
733 CMD "first derivative evaluation failed during first derivative computation.");
734
735 if (is_optimization) {
736 bool success{false};
737 FunctionOutput f; success = function.evaluate(x, f);
738 OPTIMIST_ASSERT(success,
739 CMD "function evaluation failed during first derivative computation.");
740 this->m_function_evaluations++;
741 if constexpr (FunctionInputTrait::IsScalar && FunctionOutputTrait::IsScalar) {
742 out = J*f;
743 } else if constexpr (OutputTrait::Dimension != FunctionOutputTrait::Dimension) {
744 out = J.transpose()*f;
745 } else {
746 OPTIMIST_ERROR(CMD "optimization problem inconsistent output in first derivative.");
747 }
748 } else {
749 if constexpr (OutputTrait::Dimension == FunctionOutputTrait::Dimension) {
750 out = J;
751 } else {
752 OPTIMIST_ERROR(CMD "root-finding problem with inconsistent output in first derivative.");
753 }
754 }
755 return success;
756 };
757
758 auto second_derivative_lambda = [&function, is_optimization, this] (Input const & x,
759 SecondDerivative & out) -> bool
760 {
761 bool success{false};
762 typename FunctionType::SecondDerivative H; success = function.second_derivative(x, H);
763 OPTIMIST_ASSERT(success,
764 CMD "function evaluation failed during second derivative computation.");
765
766 if (is_optimization) {
767 FunctionOutput f; success = function.evaluate(x, f);
768 this->m_function_evaluations++;
769 OPTIMIST_ASSERT(success,
770 CMD "function evaluation failed during second derivative computation.");
771 typename FunctionType::FirstDerivative J; success = function.first_derivative(x, J);
772 this->m_first_derivative_evaluations++;
773 OPTIMIST_ASSERT(success,
774 CMD "first derivative evaluation failed during second derivative computation.");
775 if constexpr (FunctionInputTrait::IsScalar && FunctionOutputTrait::IsScalar) {
776 out = J*J + f*H;
777 } else if constexpr (OutputTrait::Dimension != FunctionOutputTrait::Dimension) {
778 out = J.transpose()*J;
779 for (Integer i{0}; i < static_cast<Integer>(H.size()); ++i) {out += f(i)*H[i];}
780 }
781 else {
782 OPTIMIST_ERROR(CMD "optimization problem with inconsistent output in second derivative.");
783 }
784 } else {
785 if constexpr (OutputTrait::Dimension == FunctionOutputTrait::Dimension) {
786 out = H;
787 } else {
788 OPTIMIST_ERROR(CMD "root-finding problem with inconsistent output in second derivative.");
789 }
790 }
791 return success;
792 };

794 // Select solver method based on derivative requirements
795 if constexpr (DerivedSolver::RequiresFunction && !DerivedSolver::RequiresFirstDerivative &&
796 !DerivedSolver::RequiresSecondDerivative) {
797 return static_cast<DerivedSolver *>(this)->solve_impl(function_lambda, x_ini, x_sol);
798 } else if constexpr (DerivedSolver::RequiresFunction && DerivedSolver::RequiresFirstDerivative &&
799 !DerivedSolver::RequiresSecondDerivative) {
800 return static_cast<DerivedSolver *>(this)->solve_impl(function_lambda, first_derivative_lambda,
801 x_ini, x_sol);
802 } else if constexpr (DerivedSolver::RequiresFunction && DerivedSolver::RequiresFirstDerivative &&
803 DerivedSolver::RequiresSecondDerivative) {
804 return static_cast<DerivedSolver *>(this)->solve_impl(function_lambda, first_derivative_lambda,
805 second_derivative_lambda, x_ini, x_sol);
806 } else {
807 OPTIMIST_ERROR(CMD "no matching function signature found for the solver.");
808 }

893
894 Scalar step_norm_old, step_norm_new, residuals_old, residuals_new, tau{1.0};
895 for (this->m_relaxations = 0; this->m_relaxations < this->m_max_relaxations; ++this->m_relaxations)
896 {
897 // Update point
898 step_new = tau * step_old;
899 x_new = x_old + step_new;
900 bool success{this->evaluate_function(std::forward<FunctionLambda>(function), x_new, function_new)};
901 OPTIMIST_ASSERT(success,
902 CMD "function evaluation failed during damping.");
903
904 // Compute step norms
905 if constexpr (InputTrait::IsEigen) {
906 step_norm_old = step_old.norm();
907 step_norm_new = step_new.norm();
908 } else {
909 step_norm_old = std::abs(step_old);
910 step_norm_new = std::abs(step_new);
911 }
912
913 // Compute residual norms
914 if constexpr (OutputTrait::IsEigen) {
915 residuals_old = function_old.norm();
916 residuals_new = function_new.norm();
917 } else {
918 residuals_old = std::abs(function_old);
919 residuals_new = std::abs(function_new);
920 }
921
922 // Check relaxation
923 if (residuals_new < residuals_old || step_norm_new < (Scalar(1.0)-tau/Scalar(2.0))*step_norm_old) {
924 return true;
925 } else {
926 tau *= this->m_alpha;
927 }
928 }
929 return false;
930
931 #undef CMD
932 }
