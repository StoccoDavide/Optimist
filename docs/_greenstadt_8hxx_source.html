122 bool solve(Vector const &x_ini, Vector &x_sol) override
123 {
124 // Setup internal variables
125 this->reset();
126
127 // Print header
128 if (this->m_verbose) {this->header();}
129
130 // Initialize variables
131 Real residuals_old, residuals_new, step_norm_old, step_norm_new, tau;
132 Vector x_old, x_new, function_old, function_new, step_old, step_new, delta_x_old, delta_x_new,
133 delta_function_old, delta_function_new;
134 Matrix jacobian_old, jacobian_new;
135
136 // Set initial iteration
137 x_old = x_ini;
138 this->evaluate_function(x_old, function_old);
139 this->evaluate_jacobian(x_old, jacobian_old);
140
141 // Algorithm iterations
142 Real tolerance_residuals{this->m_tolerance};
143 Real tolerance_step_norm{this->m_tolerance * this->m_tolerance};
144 for (this->m_iterations = Integer(1); this->m_iterations < this->m_max_iterations; ++this->m_iterations)
145 {
146 // Store trace
147 this->store_trace(x_old);
148
149 // Calculate step
150 step_old = -jacobian_old * function_old;
151
152 // Check convergence
153 residuals_old = function_old.norm();
154 step_norm_old = step_old.norm();
155 if (this->m_verbose) {this->info(residuals_old);}
156 if (residuals_old < tolerance_residuals || step_norm_old < tolerance_step_norm) {
157 this->m_converged = true;
158 break;
159 }
160
161 if (this->m_damped)
162 {
163 // Relax the iteration process
164 tau = Real(1.0);
165 for (this->m_relaxations = Integer(0); this->m_relaxations < this->m_max_relaxations; ++this->m_relaxations)
166 {
167 // Update point
168 step_new = tau * step_old;
169 x_new = x_old + step_new;
170 this->evaluate_function(x_new, function_new);
171
172 // Check relaxation
173 residuals_new = function_new.norm();
174 step_norm_new = step_new.norm();
175 if (residuals_new < residuals_old || step_norm_new < (Real(1.0)-tau/Real(2.0))*step_norm_old) {
176 break;
177 } else {
178 tau *= this->m_alpha;
179 }
180 }
181 } else {
182 // Update point
183 x_new = x_old + step_old;
184 this->evaluate_function(x_new, function_new);
185 }
186
187 // Update jacobian approximation
188 delta_x_new = x_new - x_old;
189 delta_function_new = function_new - function_old;
190 this->update(
191 jacobian_old, // Old step data
192 delta_x_new, delta_function_new, function_new, jacobian_new // New step data
193 );
194
195 // Update internal variables
196 x_old = x_new;
197 function_old = function_new;
198 delta_x_old = delta_x_new;
199 delta_function_old = delta_function_new;
200 step_old = step_new;
201 jacobian_old = jacobian_new;
202 }
203
204 // Print bottom
205 if (this->m_verbose) {this->bottom();}
206
207 // Convergence data
208 x_sol = x_old;
209 return this->m_converged;
210 }

220 void update(
221 Matrix const &jacobian_old, Vector const &delta_x_new, Vector const &delta_function_new,
222 Vector const &function_new, Matrix &jacobian_new
223 ) {
224 if (this->m_method == Method::ONE) {
225 // Greenstadt's 1st method
226 // J1 = J0 - (J0*DF1-DX1)/(C'*DF1)*C', where C = F1;
227 jacobian_new = jacobian_old - (jacobian_old*delta_function_new-delta_x_new)/(function_new.transpose()*delta_function_new)*function_new.transpose();
228 } else if (this->m_method == Method::TWO) {
229 // Greenstadt's 2nd method
230 // J1 = J0 - (J0*DF1-DX1)/(C'*DF1)*C', where C = J0'*J0*DF1;
231 Vector C(jacobian_old.transpose()*jacobian_old*delta_function_new);
232 jacobian_new = jacobian_old - (jacobian_old*delta_function_new-delta_x_new)/(C.transpose()*delta_function_new)*C.transpose();
233 }
234 }
