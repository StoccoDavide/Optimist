113 bool solve_impl(Function t_function, Jacobian t_jacobian, Vector const &x_ini, Vector &x_sol)
114 {
115 // Setup internal variables
116 this->reset();
117
118 // Print header
119 if (this->m_verbose) {this->header();}
120
121 // Initialize variables
122 bool damped;
123 Real residuals, step_norm;
124 Vector x_old, x_new, function_old, function_new, step_old, step_new, delta_x_old, delta_x_new,
125 delta_function_old, delta_function_new;
126 Matrix jacobian_old, jacobian_new;
127
128 // Set initial iteration
129 x_old = x_ini;
130 this->evaluate_function(t_function, x_old, function_old);
131 this->evaluate_jacobian(t_jacobian, x_old, jacobian_old);
132
133 // Algorithm iterations
134 Real tolerance_residuals{this->m_tolerance};
135 Real tolerance_step_norm{this->m_tolerance * this->m_tolerance};
136 for (this->m_iterations = Integer(1); this->m_iterations < this->m_max_iterations; ++this->m_iterations)
137 {
138 // Store trace
139 this->store_trace(x_old);
140
141 // Calculate step
142 step_old = -jacobian_old * function_old;
143
144 // Check convergence
145 residuals = function_old.norm();
146 step_norm = step_old.norm();
147 if (this->m_verbose) {this->info(residuals);}
148 if (residuals < tolerance_residuals || step_norm < tolerance_step_norm) {
149 this->m_converged = true;
150 break;
151 }
152
153 if (this->m_damped)
154 {
155 // Relax the iteration process
156 damped = this->damp(t_function, x_old, function_old, step_old, x_new, function_new, step_new);
157 OPTIMIST_ASSERT_WARNING(damped,
158 "Optimist::RootFinder::Greenstadt::solve(...): damping failed.");
159 } else {
160 // Update point
161 x_new = x_old + step_old;
162 this->evaluate_function(t_function, x_new, function_new);
163 }
164
165 // Update jacobian approximation
166 delta_x_new = x_new - x_old;
167 delta_function_new = function_new - function_old;
168 this->update(
169 jacobian_old, // Old step data
170 delta_x_new, delta_function_new, function_new, jacobian_new // New step data
171 );
172
173 // Update internal variables
174 x_old = x_new;
175 function_old = function_new;
176 delta_x_old = delta_x_new;
177 delta_function_old = delta_function_new;
178 step_old = step_new;
179 jacobian_old = jacobian_new;
180 }
181
182 // Print bottom
183 if (this->m_verbose) {this->bottom();}
184
185 // Convergence data
186 x_sol = x_old;
187 return this->m_converged;
188 }

198 void update(
199 Matrix const &jacobian_old, Vector const &delta_x_new, Vector const &delta_function_new,
200 Vector const &function_new, Matrix &jacobian_new
201 ) {
202 if (this->m_method == Method::ONE) {
203 // Greenstadt's 1st method
204 // J1 = J0 - (J0*DF1-DX1)/(C'*DF1)*C', where C = F1;
205 jacobian_new = jacobian_old - (jacobian_old*delta_function_new-delta_x_new)/(function_new.transpose()*delta_function_new)*function_new.transpose();
206 } else if (this->m_method == Method::TWO) {
207 // Greenstadt's 2nd method
208 // J1 = J0 - (J0*DF1-DX1)/(C'*DF1)*C', where C = J0'*J0*DF1;
209 Vector C(jacobian_old.transpose()*jacobian_old*delta_function_new);
210 jacobian_new = jacobian_old - (jacobian_old*delta_function_new-delta_x_new)/(C.transpose()*delta_function_new)*C.transpose();
211 }
212 }
