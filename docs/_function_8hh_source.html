41 class FunctionBase

43 public:

150 bool is_solution(const InputType & x, const Real tol = EPSILON_LOW) const
151 {
152 for (const auto & s : this->m_solutions) {
153 if constexpr (FunInDim == 1) {
154 if (std::abs(x - s) < tol) {return true;}
155 } else if constexpr (FunInDim > 1) {
156 if((x - s).norm() < tol) {return true;}
157 } else {
158 OPTIMIST_ERROR("Optimist::FunctionBase::is_solution(...): invalid input dimension.");
159 return false;
160 }
161 }
162 return false;
163 }
