185 bool is_solution(const Input &x,
186 const Scalar tol = FunctionBase::SQRT_EPSILON) const {
187 for (const auto &s : this->m_solutions) {
188 if constexpr (InputTrait::IsEigen) {
189 if ((x - s).norm() < tol) {
190 return true;
191 }
192 } else {

194 return true;
