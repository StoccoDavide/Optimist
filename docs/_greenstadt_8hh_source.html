118 if (this->m_method == Method::ONE) {
119 // Greenstadt's 1st method
120 // J1 = J0 - (J0*DF1-DX1)/(C'*DF1)*C', where C = F1;
121 jacobian_new = jacobian_old - (jacobian_old*delta_function_new - delta_x_new) /
122 (function_new.dot(delta_function_new))*function_new.transpose();
123 } else if (this->m_method == Method::TWO) {
124 // Greenstadt's 2nd method
125 // J1 = J0 - (J0*DF1-DX1)/(C'*DF1)*C', where C = J0'*J0*DF1;
126 Vector C(jacobian_old.transpose()*jacobian_old*delta_function_new);
127 jacobian_new = jacobian_old - (jacobian_old*delta_function_new - delta_x_new) /
128 (C.dot(delta_function_new))*C.transpose();
