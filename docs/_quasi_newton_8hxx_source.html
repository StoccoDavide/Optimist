82 bool solve_impl(FunctionWrapper function, JacobianWrapper jacobian, Vector const & x_ini,
83 Vector & x_sol)
84 {
85 // Setup internal variables
86 this->reset();
87
88 // Print header
89 if (this->m_verbose) {this->header();}
90
91 // Initialize variables
92 bool damped;
93 Real residuals, step_norm;
94 Vector x_old, x_new, function_old, function_new, step_old, step_new, delta_x_old, delta_x_new,
95 delta_function_old, delta_function_new;
96 Matrix jacobian_old, jacobian_new;
97
98 // Set initial iteration
99 x_old = x_ini;
100 this->evaluate_function(function, x_old, function_old);
101 this->evaluate_jacobian(jacobian, x_old, jacobian_old);
102
103 // Algorithm iterations
104 Real tolerance_residuals{this->m_tolerance};
105 Real tolerance_step_norm{this->m_tolerance * this->m_tolerance};
106 for (this->m_iterations = Integer(1); this->m_iterations < this->m_max_iterations; ++this->m_iterations)
107 {
108 // Store trace
109 this->store_trace(x_old);
110
111 // Calculate step
112 step_old = -jacobian_old * function_old;
113
114 // Check convergence
115 residuals = function_old.norm();
116 step_norm = step_old.norm();
117 if (this->m_verbose) {this->info(residuals);}
118 if (residuals < tolerance_residuals || step_norm < tolerance_step_norm) {
119 this->m_converged = true;
120 break;
121 }
122
123 if (this->m_damped)
124 {
125 // Relax the iteration process
126 damped = this->damp(function, x_old, function_old, step_old, x_new, function_new, step_new);
127 OPTIMIST_ASSERT_WARNING(damped,
128 "Optimist::RootFinder::QuasiNewton::solve(...): damping failed.");
129 } else {
130 // Update point
131 x_new = x_old + step_old;
132 this->evaluate_function(function, x_new, function_new);
133 }
134
135 // Update jacobian approximation
136 delta_x_new = x_new - x_old;
137 delta_function_new = function_new - function_old;
138 this->update(
139 delta_x_old, delta_function_old, jacobian_old, // Old step data
140 delta_x_new, delta_function_new, function_new, jacobian_new // New step data
141 );
142
143 // Update internal variables
144 x_old = x_new;
145 function_old = function_new;
146 delta_x_old = delta_x_new;
147 delta_function_old = delta_function_new;
148 step_old = step_new;
149 jacobian_old = jacobian_new;
150 }
151
152 // Print bottom
153 if (this->m_verbose) {this->bottom();}
154
155 // Convergence data
156 x_sol = x_old;
157 return this->m_converged;
158 }
