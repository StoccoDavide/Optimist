79 bool solve_impl(FunctionWrapper function, JacobianWrapper jacobian, Vector const &x_ini,
80 Vector &x_sol)
81 {
82 // Setup internal variables
83 this->reset();
84
85 // Print header
86 if (this->m_verbose) {this->header();}
87
88 // Initialize variables
89 bool damped;
90 Real residuals, step_norm;
91 Vector x_old, x_new, function_old, function_new, step_old, step_new, delta_x_old, delta_x_new,
92 delta_function_old, delta_function_new;
93 Matrix jacobian_old, jacobian_new;
94
95 // Set initial iteration
96 x_old = x_ini;
97 this->evaluate_function(function, x_old, function_old);
98 this->evaluate_jacobian(jacobian, x_old, jacobian_old);
99
100 // Algorithm iterations
101 Real tolerance_residuals{this->m_tolerance};
102 Real tolerance_step_norm{this->m_tolerance * this->m_tolerance};
103 for (this->m_iterations = Integer(1); this->m_iterations < this->m_max_iterations; ++this->m_iterations)
104 {
105 // Store trace
106 this->store_trace(x_old);
107
108 // Calculate step
109 step_old = -jacobian_old * function_old;
110
111 // Check convergence
112 residuals = function_old.norm();
113 step_norm = step_old.norm();
114 if (this->m_verbose) {this->info(residuals);}
115 if (residuals < tolerance_residuals || step_norm < tolerance_step_norm) {
116 this->m_converged = true;
117 break;
118 }
119
120 if (this->m_damped)
121 {
122 // Relax the iteration process
123 damped = this->damp(function, x_old, function_old, step_old, x_new, function_new, step_new);
124 OPTIMIST_ASSERT_WARNING(damped,
125 "Optimist::RootFinder::QuasiNewton::solve(...): damping failed.");
126 } else {
127 // Update point
128 x_new = x_old + step_old;
129 this->evaluate_function(function, x_new, function_new);
130 }
131
132 // Update jacobian approximation
133 delta_x_new = x_new - x_old;
134 delta_function_new = function_new - function_old;
135 this->update(
136 delta_x_old, delta_function_old, jacobian_old, // Old step data
137 delta_x_new, delta_function_new, function_new, jacobian_new // New step data
138 );
139
140 // Update internal variables
141 x_old = x_new;
142 function_old = function_new;
143 delta_x_old = delta_x_new;
144 delta_function_old = delta_function_new;
145 step_old = step_new;
146 jacobian_old = jacobian_new;
147 }
148
149 // Print bottom
150 if (this->m_verbose) {this->bottom();}
151
152 // Convergence data
153 x_sol = x_old;
154 return this->m_converged;
155 }
