72 template <typename FunctionLambda, typename FirstDerivativeLambda, typename SecondDerivativeLambda>
73 bool solve_impl(FunctionLambda && function, FirstDerivativeLambda && first_derivative,
74 SecondDerivativeLambda && second_derivative, Scalar x_ini, Scalar & x_sol)
75 {
76 #define CMD "Optimist::RootFinder::Halley::solve(...): "
77
78 // Reset internal variables
79 this->reset_counters();
80
81 // Print header
82 if (this->m_verbose) {this->header();}
83
84 // Initialize variables
85 bool damped, success;
86 Scalar residuals, step_norm;
87 Scalar x_old, x_new, function_old, function_new, step_old, step_new;
88 Scalar first_derivative_old, second_derivative_old;
89
90 // Set initial iteration
91 x_old = x_ini;
92 success = this->evaluate_function(std::forward<FunctionLambda>(function), x_old, function_old);
93 OPTIMIST_ASSERT(success,
94 CMD "function evaluation failed at the initial point.");
95
96 // Algorithm iterations
97 Scalar tolerance_residuals{this->m_tolerance};
98 Scalar tolerance_step_norm{this->m_tolerance * this->m_tolerance};
99 for (this->m_iterations = 1; this->m_iterations < this->m_max_iterations; ++this->m_iterations)
100 {
101 // Evaluate derivatives
102 success = this->evaluate_first_derivative(std::forward<FirstDerivativeLambda>(first_derivative), x_old, first_derivative_old);
103 OPTIMIST_ASSERT(success,
104 CMD "first derivative evaluation failed at iteration " << this->m_iterations << ".");
105 success = this->evaluate_second_derivative(std::forward<SecondDerivativeLambda>(second_derivative), x_old, second_derivative_old);
106 OPTIMIST_ASSERT(success,
107 CMD "second derivative evaluation failed at iteration " << this->m_iterations << ".");
108
109 // Calculate step
110 if (std::abs(first_derivative_old) < Halley::SQRT_EPSILON) {
111 OPTIMIST_WARNING(CMD "close-to-singular first derivative detected.");
112 first_derivative_old = (first_derivative_old > 0) ? Halley::SQRT_EPSILON : -Halley::SQRT_EPSILON;
113 }
114 if (std::abs(second_derivative_old) < Halley::SQRT_EPSILON) {
115 OPTIMIST_WARNING(CMD "close-to-singular second derivative detected.");
116 second_derivative_old = (second_derivative_old > 0) ? Halley::SQRT_EPSILON : -Halley::SQRT_EPSILON;
117 }
118 step_old = -(function_old / first_derivative_old) /
119 (1.0 - (function_old * second_derivative_old) / (2.0 * first_derivative_old * first_derivative_old));
120 OPTIMIST_ASSERT(std::isfinite(step_old), CMD "step " << this->m_iterations << " is not finite.");
121
122 // Check convergence
123 residuals = std::abs(function_old);
124 step_norm = std::abs(step_old);
125 if (this->m_verbose) {this->info(residuals);}
126 if (residuals < tolerance_residuals || step_norm < tolerance_step_norm) {
127 this->m_converged = true;
128 break;
129 }
130
131 if (this->m_damped) {
132 // Relax the iteration process
133 damped = this->damp(std::forward<FunctionLambda>(function), x_old, function_old, step_old, x_new, function_new, step_new);
134 OPTIMIST_ASSERT_WARNING(damped, CMD "damping failed.");
135 } else {
136 // Update point
137 x_new = x_old + step_old;
138 success = this->evaluate_function(std::forward<FunctionLambda>(function), x_new, function_new);
139 OPTIMIST_ASSERT(success,
140 CMD "function evaluation failed at iteration " << this->m_iterations << ".");
141 }
142
143 // Update internal variables
144 x_old = x_new;
145 function_old = function_new;
146 step_old = step_new;
147 }
148
149 // Print bottom
150 if (this->m_verbose) {this->bottom();}
151
152 // Convergence data
153 x_sol = x_old;
154 return this->m_converged;
155
156 #undef CMD
157 }
