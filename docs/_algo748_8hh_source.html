86 template <typename FunctionLambda>
87 bool bracketing(FunctionLambda && function)
88 {
89 #define CMD "Optimist::RootFinder::Algo748::bracketing(...): "
90
91 {
92 Scalar tolerance{static_cast<Scalar>(0.7)*this->m_tolerance_bracketing};
93 Scalar hba{(this->m_b - this->m_a)/static_cast<Scalar>(2.0)};
94 if (hba <= tolerance) {this->m_c = this->m_a + hba;}
95 else if (this->m_c <= this->m_a + tolerance) {this->m_c = this->m_a + tolerance;}
96 else if (this->m_c >= this->m_b - tolerance) {this->m_c = this->m_b - tolerance;}
97 }
98
99 // If f(c) = 0 set a = c and return true
100 bool success{this->evaluate_function(std::forward<FunctionLambda>(function), this->m_c, this->m_fc)};
101 OPTIMIST_ASSERT(success,
102 CMD "function evaluation failed during bracketing.");
103 if (this->m_fc == 0) {
104 this->m_a = this->m_c; this->m_fa = 0.0;
105 this->m_d = 0.0; this->m_fd = 0.0;
106 return true;
107 } else {
108 // If f(c) is not zero, then determine the new enclosing interval
109 if (this->m_fa*this->m_fc < 0) {
110 // D <- B <- C
111 this->m_d = this->m_b; this->m_fd = this->m_fb;
112 this->m_b = this->m_c; this->m_fb = this->m_fc;
113 } else {
114 // D <- A <- C
115 this->m_d = this->m_a; this->m_fd = this->m_fa;
116 this->m_a = this->m_c; this->m_fa = this->m_fc;
117 }
118 return false;
119 }

129 Scalar cubic_interpolation()
130 {
131 // Compute the coefficients for the inverse cubic interpolation
132 Scalar d_1{this->m_b - this->m_a};
133 Scalar d_2{this->m_d - this->m_a};
134 Scalar d_3{this->m_e - this->m_a};
135
136 Scalar dd_0{d_1/(this->m_fb - this->m_fa)};
137 Scalar dd_1{(d_1-d_2)/(this->m_fb - this->m_fd)};
138 Scalar dd_2{(d_2-d_3)/(this->m_fd - this->m_fe)};
139
140 Scalar ddd_0 {(dd_0 - dd_1)/(this->m_fa - this->m_fd)};
141 Scalar ddd_1 {(dd_1 - dd_2)/(this->m_fb - this->m_fe)};
142
143 Scalar dddd_0{(ddd_0-ddd_1)/(this->m_fa-this->m_fe)};
144
145 // Compute the approximate root
146 Scalar root{this->m_a - this->m_fa*(dd_0 - this->m_fb*(ddd_0 - this->m_fd*dddd_0))};
147 Scalar tolerance{static_cast<Scalar>(0.7)*this->m_tolerance_bracketing};
148 if (root <= this->m_a + tolerance || root >= this->m_b - tolerance) {
149 root = (this->m_a + this->m_b)/2.0;
150 }
151 return root;
152 }

161 Scalar newton_quadratic(Integer n)
162 {
163 // Compute the coefficients for the quadratic interpolation
164 Scalar a_0{this->m_fa};
165 Scalar a_1{(this->m_fb - this->m_fa)/(this->m_b - this->m_a)};
166 Scalar a_2{((this->m_fd - this->m_fb)/(this->m_d - this->m_b)-a_1)/(this->m_d - this->m_a)};
167
168 // Safeguard to avoid overflow
169 if (a_2 == 0) {return this->m_a - a_0/a_1;}
170
171 // Determine the starting point of newton steps
172 Scalar c{a_2*this->m_fa > 0 ? this->m_a : this->m_b};
173
174 // Start the safeguarded newton steps
175 bool is_nonzero{true};
176 Scalar pc{0.0}, pdc{0.0};
177 for (Integer i{0}; i < n && is_nonzero ; ++i ) {
178 pc = a_0 + (a_1 + a_2*(c - this->m_b))*(c - this->m_a);
179 pdc = a_1 + a_2*((2.0 * c) - (this->m_a + this->m_b));
180 is_nonzero = pdc != 0;
181 if (is_nonzero) {c -= pc/pdc;}
182 }
183
184 if (is_nonzero) {return c;}
185 else {return this->m_a - a_0/a_1;}
186 }

202 template <typename FunctionLambda>
203 Scalar find_root_impl(FunctionLambda && function)
204 {
205 #define CMD "Optimist::RootFinder::Algo748::find_root_impl(...): "
206
207 bool success;
208 Scalar tolerance_step{this->m_tolerance_bracketing};
209 Scalar tolerance_function{this->m_tolerance_bracketing};
210
211 // Check for trivial solution
212 this->m_converged = this->m_fa == 0;
213 if (this->m_verbose) {this->info(std::abs(this->m_fa));}
214 if (this->m_converged) {return this->m_fa;}
215 this->m_converged = this->m_fb == 0;
216 if (this->m_verbose) {this->info(std::abs(this->m_fb));}
217 if (this->m_converged) {return this->m_fb;}
218
219 // Initialize to dumb values
220 this->m_e = QUIET_NAN;
221 this->m_fe = QUIET_NAN;
222
223 // While f(left) or f(right) are infinite perform bisection
224 while (!(std::isfinite(this->m_fa) && std::isfinite(this->m_fb))) {
225 ++this->m_iterations;
226 this->m_c = (this->m_a + this->m_b)/2.0;
227 success = this->evaluate_function(std::forward<FunctionLambda>(function), this->m_c, this->m_fc);
228 OPTIMIST_ASSERT(success,
229 CMD "function evaluation failed at iteration " << this->m_iterations << ".");
230 this->m_converged = this->m_fc == 0;
231 if (this->m_verbose) {this->info(std::abs(this->m_fc));}
232 if (this->m_converged) {return this->m_fc;}
233 if (this->m_fa*this->m_fc < 0) {
234 // -> [a, c]
235 this->m_b = this->m_c; this->m_fb = this->m_fc;
236 } else {
237 // -> [c, b]
238 this->m_a = this->m_c; this->m_fa = this->m_fc;
239 }
240
241 // Check for convergence
242 Scalar abs_fa{std::abs(this->m_fa)};
243 Scalar abs_fb{std::abs(this->m_fb)};
244 if (this->m_verbose) {this->info(abs_fa < abs_fb ? abs_fa : abs_fb);}
245 this->m_converged = (this->m_b - this->m_a) < tolerance_step ||
246 abs_fa < tolerance_function ||abs_fb < tolerance_function;
247 if (this->m_converged) {return abs_fb < abs_fa ? this->m_b : this->m_a;}
248 }
249
250 {
251 // Impede the step c to be too close to the bounds a or b.
252 Scalar b_a{this->m_b - this->m_a};
253 Scalar r{b_a/(this->m_fb - this->m_fa)};
254 this->m_c = std::abs(this->m_fb) < std::abs(this->m_fa) ?
255 this->m_b + this->m_fb*r : this->m_a - this->m_fa*r;
256 Scalar delta{this->m_interval_shink*b_a}, a_min, b_max;
257 if (this->m_c < (a_min = this->m_a + delta)) {this->m_c = a_min;}
258 else if (this->m_c > (b_max = this->m_b - delta)) {this->m_c = b_max;}
259 }
260
261 // Call "bracketing" to get a shrinked enclosing interval and to update the termination criterion
262 this->m_converged = this->bracketing(std::forward<FunctionLambda>(function));
263 if (this->m_verbose) {this->info(std::abs(this->m_fa));}
264 if (this->m_converged) {return this->m_a;}
265 this->m_converged = false;
266
267 // The enclosing interval is recorded as [a0, b0] before executing the iteration steps
268 while (++this->m_iterations < this->m_max_iterations) {
269 ++this->m_iterations;
270 Scalar a0_b0{this->m_b - this->m_a};
271
272 // Check the termination criterion
273 {
274 Scalar abs_fa {std::abs(this->m_fa)};
275 Scalar abs_fb {std::abs(this->m_fb)};
276 if (this->m_verbose) {this->info(abs_fa < abs_fb ? abs_fa : abs_fb);}
277 this->m_converged = abs_fa < tolerance_function || abs_fb < tolerance_function;
278 if (this->m_converged) {return abs_fa < abs_fb ? this->m_a : this->m_b;}
279 }
280
281 // Starting with the second iteration: newton quadratic or cubic interpolation
282 bool do_newton_quadratic{false};
283 if (!std::isfinite(this->m_fe)) {
284 do_newton_quadratic = true;
285 } else if (!this->all_different( this->m_fa, this->m_fb, this->m_fd, this->m_fe)) {
286 do_newton_quadratic = true;
287 } else {
288 this->m_c = this->cubic_interpolation();
289 do_newton_quadratic = (this->m_c-this->m_a)*(this->m_c-this->m_b) >= 0.0;
290 }
291 if (do_newton_quadratic) {this->m_c = this->newton_quadratic(2);}
292
293 this->m_e = this->m_d;
294 this->m_fe = this->m_fd;
295
296 // Call "bracketing" to get a shrinked enclosing interval and to update the termination criterion
297 this->m_converged = this->bracketing(std::forward<FunctionLambda>(function)) || (this->m_b-this->m_a) <= this->m_tolerance;
298 if (this->m_verbose) {this->info(std::abs(this->m_fa) < std::abs(this->m_fb) ? std::abs(this->m_fa) : std::abs(this->m_fb));}
299 if (this->m_converged) {
300 return std::abs(this->m_fa) < std::abs(this->m_fb) ? this->m_a : this->m_b;
301 }
302 if (!this->all_different(this->m_fa, this->m_fb, this->m_fd, this->m_fe)) {
303 do_newton_quadratic = true;
304 } else {
305 this->m_c = this->cubic_interpolation();
306 do_newton_quadratic = (this->m_c - this->m_a)*(this->m_c - this->m_b) >= 0.0;
307 }
308 if (do_newton_quadratic) {this->m_c = this->newton_quadratic(3);}
309
310 // Call "bracketing" to get a shrinked enclosing interval and to update the termination criterion
311 {
312 this->m_converged = this->bracketing(std::forward<FunctionLambda>(function)) || (this->m_b-this->m_a) <= this->m_tolerance;
313 Scalar abs_fa{std::abs(this->m_fa)};
314 Scalar abs_fb{std::abs(this->m_fb)};
315 if (this->m_verbose) {this->info(abs_fa < abs_fb ? abs_fa : abs_fb);}
316 if (this->m_converged) {return abs_fa < abs_fb ? this->m_a : this->m_b;}
317
318 this->m_e = this->m_d;
319 this->m_fe = this->m_fd;
320
321 // Takes the double-size secant step
322 Scalar u, fu;
323 if (abs_fa < abs_fb) {u = this->m_a; fu = this->m_fa;}
324 else {u = this->m_b; fu = this->m_fb;}
325 Scalar hba{(this->m_b - this->m_a)/static_cast<Scalar>(2.0)};
326 this->m_c = u - 4.0*(fu/(this->m_fb - this->m_fa))*hba;
327 if (std::abs(this->m_c - u) > hba) {this->m_c = this->m_a + hba;}
328 }
329
330 // Call "bracketing" to get a shrinked enclosing interval and to update the termination criterion
331 this->m_converged = this->bracketing(std::forward<FunctionLambda>(function)) || (this->m_b-this->m_a) <= this->m_tolerance_bracketing;
332 if (this->m_converged) {
333 return std::abs(this->m_fa) < std::abs(this->m_fb) ? this->m_a : this->m_b;
334 }
335
336 // Determine whether an additional bisection step is needed
337 if ((this->m_b-this->m_a) < this->m_mu*a0_b0) {continue;}
338
339 this->m_e = this->m_d;
340 this->m_fe = this->m_fd;
341
342 // Call "bracketing" to get a shrinked enclosing interval and to update the termination criterion
343 {
344 Scalar b_a{this->m_b-this->m_a};
345 this->m_c = this->m_a + b_a/2.0;
346 this->m_converged = this->bracketing(std::forward<FunctionLambda>(function)) || b_a <= this->m_tolerance_bracketing;
347 }
348 }
349 // Return the approximate root
350 return std::abs(this->m_fa) < std::abs(this->m_fb) ? this->m_a : this->m_b;
