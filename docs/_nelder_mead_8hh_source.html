267 template <typename FunctionLambda>
268 bool solve_impl(FunctionLambda && function, Vector const & x_ini, Vector & x_sol)
269 {
270 #define CMD "Optimist::Optimizer::NelderMead::solve_impl(...): "
271
272 // Precompute constants
273 this->m_dim = x_ini.size();
274 this->m_dim_factorial = 1;
275 for (Integer i{2}; i <= this->m_dim+1; ++i) {this->m_dim_factorial *= i;}
276 this->m_regular_simplex_volume = (std::sqrt(static_cast<Scalar>(this->m_dim+1)) /
277 static_cast<Scalar>(this->m_dim_factorial)) / std::pow(2.0, static_cast<Scalar>(this->m_dim)/2.0);
278
279 // Initialize internal variables
280 if constexpr (VectorTrait::IsFixed) {
281 this->m_p.setZero();
282 this->m_dist.setZero();
283 this->m_f.setZero();
284 this->m_f_work.setZero();
285 this->m_p_work.setZero();
286 this->m_p_sum.setZero();
287 this->m_p_r.setZero();
288 this->m_p_e.setZero();
289 this->m_p_c.setZero();
290 } else if constexpr (VectorTrait::IsDynamic) {
291 this->m_p.resize(this->m_dim, this->m_dim+1);
292 this->m_dist.resize(this->m_dim+1, this->m_dim+1);
293 this->m_f.resize(this->m_dim+1);
294 this->m_f_work.resize(this->m_dim+1);
295 this->m_p_work.resize(this->m_dim, this->m_dim+1);
296 this->m_p_sum.resize(this->m_dim);
297 this->m_p_r.resize(this->m_dim);
298 this->m_p_e.resize(this->m_dim);
299 this->m_p_c.resize(this->m_dim);
300 this->m_p.setZero();
301 this->m_dist.setZero();
302 this->m_f.setZero();
303 this->m_f_work.setZero();
304 this->m_p_work.setZero();
305 this->m_p_sum.setZero();
306 this->m_p_r.setZero();
307 this->m_p_e.setZero();
308 this->m_p_c.setZero();
309 } else if constexpr (VectorTrait::IsSparse) {
310 this->m_p.resize(this->m_dim, this->m_dim+1);
311 this->m_dist.resize(this->m_dim+1, this->m_dim+1);
312 this->m_f.resize(this->m_dim+1);
313 this->m_f_work.resize(this->m_dim+1);
314 this->m_p_work.resize(this->m_dim, this->m_dim+1);
315 this->m_p_sum.resize(this->m_dim);
316 this->m_p_r.resize(this->m_dim);
317 this->m_p_e.resize(this->m_dim);
318 this->m_p_c.resize(this->m_dim);
319 this->m_p.reserve(this->m_dim, this->m_dim+1);
320 this->m_dist.reserve(this->m_dim+1, this->m_dim+1);
321 this->m_f.reserve(this->m_dim+1);
322 this->m_f_work.reserve(this->m_dim+1);
323 this->m_p_work.reserve(this->m_dim, this->m_dim+1);
324 this->m_p_sum.reserve(this->m_dim);
325 this->m_p_r.reserve(this->m_dim);
326 this->m_p_e.reserve(this->m_dim);
327 this->m_p_c.reserve(this->m_dim);
328 }
329
330 // Initialize simplex
331 if (this->m_method == Method::STANDARD) {
332 this->diamond(std::forward<FunctionLambda>(function), x_ini, this->m_delta);
333 } else if (this->m_method == Method::SPENDLEY) {
334 this->spendley(std::forward<FunctionLambda>(function), x_ini, this->m_delta);
335 } else {
336 OPTIMIST_ERROR(CMD "invalid method detected.");
337 }
338
339 // Reset internal variables
340 this->reset_counters();
341
342 // Print header
343 if (this->m_verbose) {this->header();}
344
345 this->dist_init();
346 this->m_p_sum.noalias() = this->m_p.col(this->m_dim);
347 for (Integer i{0}; i < this->m_dim; ++i) {this->m_p_sum.noalias() += this->m_p.col(i);}
348
349 // Algorithm iterations
350 Move move{Move::INITIALIZE};
351 for (this->m_iterations = 1; this->m_iterations <= this->m_max_iterations; ++this->m_iterations)
352 {
353 // Determine which point is the highest (worst), next-highest, and lowest (best).
354 this->m_low = 0;
355 this->m_mid = 1;
356 this->m_high = 2;
357 if (this->m_f(this->m_low) > this->m_f(this->m_mid)) {std::swap(this->m_low, this->m_mid);}
358 if (this->m_f(this->m_low) > this->m_f(this->m_high)) {std::swap(this->m_low, this->m_high);}
359 if (this->m_f(this->m_mid) > this->m_f(this->m_high)) {std::swap(this->m_mid, this->m_high);}
360 for (Integer i{3}; i <= this->m_dim; ++i)
361 {
362 if (this->m_f(i) < this->m_f(this->m_low) ) {
363 this->m_low = i; // New minima
364 } else if (this->m_f(i) > this->m_f(this->m_high) ) {
365 this->m_mid = this->m_high;
366 this->m_high = i;
367 } else if (this->m_f(i) > this->m_f(this->m_mid) ) {
368 this->m_mid = i;
369 }
370 }
371
372 Scalar f_low{this->m_f(this->m_low)};
373 Scalar f_mid{this->m_f(this->m_mid)};
374 Scalar f_high{this->m_f(this->m_high)};
375
376 // Compute the fractional range from highest to lowest and return if satisfactory
377 Scalar rtol{std::abs(f_high - f_low) / (std::abs(f_low) + this->m_tolerance)};
378
379 if (this->m_verbose) {
380 std::string move_str;
381 switch (move) {
382 case Move::INITIALIZE: move_str = "Initialize"; break;
383 case Move::REFLECT: move_str = "Reflect"; break;
384 case Move::EXPAND_E: move_str = "Expand"; break;
385 case Move::EXPAND_R: move_str = "Expand (reflection)"; break;
386 case Move::CONTRACT_O: move_str = "Contract (outside)"; break;
387 case Move::CONTRACT_I: move_str = "Contract (inside)"; break;
388 case Move::SHRINK: move_str = "Shrink"; break;
389 case Move::RESTART: move_str = "Restart"; break;
390 default: move_str = "Unknown"; break;
391 }
392 this->info(rtol, move_str);
393 }
394 if (rtol < this->m_tolerance || this->m_diameter < this->m_tolerance) {
395 this->m_converged = true;
396 break;
397 }
398
399 Scalar ratio{std::pow(this->m_simplex_volume/this->m_regular_simplex_volume, 1.0/this->m_dim) /
400 this->m_diameter};
401 if (ratio <= this->m_volume_tolerance)
402 {
403 if (this->m_method == Method::STANDARD) {
404 this->diamond(std::forward<FunctionLambda>(function), this->m_p.col(this->m_low),
405 this->m_diameter*this->m_volume_tolerance);
406 } else if (this->m_method == Method::SPENDLEY) {
407 this->spendley(std::forward<FunctionLambda>(function), this->m_p.col(this->m_low),
408 this->m_diameter*this->m_volume_tolerance);
409 } else {
410 OPTIMIST_ERROR(CMD "invalid method detected.");
411 }
412 this->dist_init();
413 this->m_p_sum.noalias() = this->m_p.col(this->m_dim);
414 for (Integer i{0}; i < this->m_dim; ++i) {this->m_p_sum.noalias() += this->m_p.col(i);}
415 move = Move::RESTART;
416 continue;
417 }
418
419 // Begin a new iteration: first extrapolate by a factor alpha (default: −1 a reflection)
420 // through the face of the simplex across from the high point, i.e., reflect the simplex
421 // from the high point
422 Scalar f_r{this->reflect(std::forward<FunctionLambda>(function), this->m_p_r)};
423 if (f_r < f_low) {
424 Scalar f_e{this->expand(std::forward<FunctionLambda>(function), this->m_p_e)};
425 if (f_e < f_r) {
426 move = Move::EXPAND_E;
427 this->replace_point(f_e, this->m_p_e, this->m_high);
428 } else {
429 move = Move::EXPAND_R;
430 this->replace_point(f_r, this->m_p_r, this->m_high);
431 }
432 } else if (f_r < f_mid) {
433 move = Move::REFLECT;
434 this->replace_point(f_r, this->m_p_r, this->m_high);
435 } else if (f_r < f_high) {
436 Scalar f_c{this->outside(std::forward<FunctionLambda>(function), this->m_p_c)};
437 if (f_c < f_r) {
438 move = Move::CONTRACT_O;
439 this->replace_point(f_c, this->m_p_c, this->m_high);
440 } else {
441 move = Move::REFLECT;
442 this->replace_point(f_r, this->m_p_r, this->m_high);
443 }
444 } else {
445 Scalar f_c{this->inside(std::forward<FunctionLambda>(function), this->m_p_c)};
446 if (f_c < f_high) {
447 move = Move::CONTRACT_I;
448 this->replace_point(f_c, this->m_p_c, this->m_high);
449 } else {
450 move = Move::SHRINK;
451 this->shrink(std::forward<FunctionLambda>(function));
452 }
453 }
454 }
455
456 // Print bottom
457 if (this->m_verbose) {this->bottom();}
458
459 // Convergence data
460 x_sol = this->m_p.col(this->m_low);
461 return this->m_converged;
462
463 #undef CMD
464 }

473 void replace_point(Scalar const f, Vector const & p, Integer const j)
474 {
475 this->m_f(j) = f;
476 this->m_p_sum += p - this->m_p.col(j);
477 this->m_p.col(j).noalias() = p;
478 this->dist_update(j);
479 }

488 template <typename FunctionLambda>
489 void spendley(FunctionLambda && function, Vector const x, Scalar const delta)
490 {
491 #define CMD "Optimist::Optimizer::NelderMead::spendley(...): "
492
493 Scalar const t1{std::sqrt(static_cast<Scalar>(this->m_dim + 1)) - 1.0};
494 Scalar const t2{static_cast<Scalar>(this->m_dim)*std::sqrt(static_cast<Scalar>(2))};
495 Scalar const p{delta * (this->m_dim+t1)/t2};
496 Scalar const q{delta * t1/t2};
497 bool success;
498 for (Integer i{0}; i < this->m_dim; ++i) {this->m_p(i,0) = x(i);}
499 success = this->evaluate_function(std::forward<FunctionLambda>(function), this->m_p.col(0), this->m_f(0));
500 OPTIMIST_ASSERT(success,
501 CMD "function evaluation failed at the initial point.");
502 for (Integer i{0}; i < this->m_dim; ++i) {
503 this->m_p.col(i+1) = this->m_p.col(0).array()+p;
504 this->m_p(i, i+1) += q;
505 success = this->evaluate_function(std::forward<FunctionLambda>(function), this->m_p.col(i+1), this->m_f(i+1));
506 OPTIMIST_ASSERT(success,
507 CMD "function evaluation failed at simplex point " << i+1 << ".");
508 }
509
510 #undef CMD
511 }

520 template <typename FunctionLambda>
521 void diamond(FunctionLambda && function, Vector const x, Scalar const delta)
522 {
523 #define CMD "Optimist::Optimizer::NelderMead::diamond(...): "
524
525 bool success;
526 for (Integer i{0}; i < this->m_dim; ++i) {this->m_p.row(i).setConstant(x(i));}
527 success = this->evaluate_function(std::forward<FunctionLambda>(function), x, this->m_f(0));
528 OPTIMIST_ASSERT(success,
529 CMD "function evaluation failed at the initial point.");
530 for (Integer i{0}; i < this->m_dim; ++i)
531 {
532 Scalar f_p, f_m;
533 this->m_p(i, i+1) = x(i) + delta;
534 success = this->evaluate_function(std::forward<FunctionLambda>(function), this->m_p.col(i+1), f_p);
535 OPTIMIST_ASSERT(success,
536 CMD "function evaluation failed at simplex point " << i+1 << ".");
537 this->m_p(i, i+1) = x(i) - delta;
538 success = this->evaluate_function(std::forward<FunctionLambda>(function), this->m_p.col(i+1), f_m);
539 OPTIMIST_ASSERT(success,
540 CMD "function evaluation failed at simplex point " << i+1 << ".");
541 if (f_p < f_m) {
542 this->m_f(i+1) = f_p;
543 this->m_p(i, i+1) = x(i) + delta;
544 } else {
545 this->m_f(i+1) = f_m;
546 }
547 }
548
549 #undef CMD
550 }

556 void simplex_volume(Integer k)
557 {
558 Integer j{0};
559 for (Integer i{0}; i <= this->m_dim; ++i)
560 {
561 if (i != k) {
562 this->m_p_work.col(j).noalias() = this->m_p.col(i) - this->m_p.col(k);
563 this->m_f_work(j) = this->m_f(i) - this->m_f(k);
564 ++j;
565 }
566 }
567 this->m_lu.compute(this->m_p_work.transpose());
568 this->m_simplex_volume = std::abs(this->m_lu.determinant()) / this->m_dim_factorial;
569 }
570
574 void dist_init()
575 {
576 for (Integer i{0}; i < this->m_dim; ++i) {
577 this->m_dist(i, i) = 0;
578 for (Integer j{i+1}; j <= this->m_dim; ++j) {
579 this->m_dist(i, j) = (this->m_p.col(i) - this->m_p.col(j)).norm();
580 this->m_dist(j, i) = this->m_dist(i, j);
581 }
582 }
583 this->m_diameter = this->m_dist.maxCoeff();
584 this->simplex_volume(0);
585 }
586
591 void dist_update(Integer j)
592 {
593 for (Integer i{0}; i <= this->m_dim; ++i) {
594 if (i != j) {
595 this->m_dist(i, j) = (this->m_p.col(i) - this->m_p.col(j) ).norm();
596 this->m_dist(j, i) = this->m_dist(i, j);
597 }
598 }
599 this->m_diameter = this->m_dist.maxCoeff();
600 this->simplex_volume(this->m_low);
601 }

608 template <typename FunctionLambda>
609 void shrink(FunctionLambda && function)
610 {
611 #define CMD "Optimist::Optimizer::NelderMead::shrink(...): "
612
613 bool success;
614 Scalar const c_1{static_cast<Scalar>(1.0) - this->m_sigma}, c_2{this->m_sigma};
615 for (Integer i{0}; i <= this->m_dim; ++i)
616 {
617 if (i != this->m_low) {
618 this->m_p.col(i) = c_1 * this->m_p.col(i) + c_2 * this->m_p.col(this->m_low);
619 success = this->evaluate_function(std::forward<FunctionLambda>(function), this->m_p.col(i), this->m_f(i));
620 OPTIMIST_ASSERT(success,
621 CMD "function evaluation failed at simplex point " << i << ".");
622 }
623 }
624 this->m_p_sum.noalias() = this->m_p.col(this->m_dim);
625 for (Integer i{0}; i < this->m_dim; ++i) {this->m_p_sum.noalias() += this->m_p.col(i);}
626 this->m_dist *= c_1;
627 this->m_diameter *= c_1;
628 this->m_simplex_volume *= std::pow(c_1, static_cast<Scalar>(this->m_dim));
629 #undef CMD
630 }

641 template <typename FunctionLambda>
642 Scalar extrapolate(FunctionLambda && function, Scalar const alpha, Integer const j, Vector & p)
643 {
644 // Extrapolates by a factor alpha through the face of the simplex
645 // across from the high point.
646 //
647 // pface = (sum(p)-pfrom)/N
648 // pcenter = sum(p)/(N+1);
649 // pe = pface + alpha*(pface-pfrom)
650 // = pface*(1+alpha)-pfrom*alpha
651 // = (sum(p)-pfrom)/N*(1+alpha)-pfrom*alpha
652 // = sum(p)/N*(1+alpha)-pfrom*(alpha+(1+alpha)/N)
653 // = sum(p)/(N+1) * (1+alpha)*((N+1)/N) - pfrom*[1/N+alpha*(N+1)/N]
654 // = pcenter * (1+alpha)*((N+1)/N) - pfrom*(1/N-(N+1)/N+(1+alpha)*(N+1)/N)
655 // = pcenter * (1+alpha)*((N+1)/N) - pfrom*(-1+(1+alpha)*(N+1)/N)
656 //
657 // let beta = (1+alpha)*((N+1)/N);
658 //
659 // pe = pcenter * beta + pfrom * (1-beta)
660
661 #define CMD "Optimist::Optimizer::NelderMead::extrapolate(...): "
662
663 Scalar const beta_1{(1.0 + alpha)/static_cast<Scalar>(this->m_n)};
664 Scalar const beta{(static_cast<Scalar>(this->m_n) + 1.0)*beta_1};
665 p.noalias() = beta_1*this->m_p_sum + (1.0 - beta) * this->m_p.col(j);
666 Scalar f;
667 bool success{this->evaluate_function(std::forward<FunctionLambda>(function), p, f)};
668 OPTIMIST_ASSERT(success,
669 CMD "function evaluation failed during extrapolation.");
670 return f;
671
672 #undef CMD
673 }
