116 template <typename FunctionLambda, typename FirstDerivativeLambda>
117 bool solve_impl(FunctionLambda && function, FirstDerivativeLambda && first_derivative, Scalar x_ini,
118 Scalar & x_sol)
119 {
120 #define CMD "Optimist::RootFinder::Varona::solve(...): "
121
122 // Reset internal variables
123 this->reset_counters();
124
125 // Print header
126 if (this->m_verbose) {this->header();}
127
128 // Initialize variables
129 bool damped, success;
130 Scalar residuals, step_norm;
131 Scalar x_old, x_new, function_old, function_new, step_old, step_new;
132 Scalar first_derivative_old;
133
134 // Set initial iteration
135 x_old = x_ini;
136 success = this->evaluate_function(std::forward<FunctionLambda>(function), x_old, function_old);
137 OPTIMIST_ASSERT(success,
138 CMD "function evaluation failed at the initial point.");
139
140 // Algorithm iterations
141 Scalar tolerance_residuals{this->m_tolerance};
142 Scalar tolerance_step_norm{this->m_tolerance * this->m_tolerance};
143 for (this->m_iterations = 1; this->m_iterations < this->m_max_iterations; ++this->m_iterations)
144 {
145 // Evaluate first derivative
146 success = this->evaluate_first_derivative(std::forward<FirstDerivativeLambda>(first_derivative), x_old, first_derivative_old);
147 OPTIMIST_ASSERT(success,
148 CMD "first derivative evaluation failed at iteration " << this->m_iterations << ".");
149
150 // Calculate step
151 if (std::abs(first_derivative_old) < Varona::SQRT_EPSILON) {
152 OPTIMIST_WARNING(CMD "close-to-singular first derivative detected.");
153 first_derivative_old = (first_derivative_old > 0) ? Varona::SQRT_EPSILON : -Varona::SQRT_EPSILON;
154 }
155
156 this->compute_step(std::forward<FunctionLambda>(function), x_old, function_old, first_derivative_old, step_old);
157
158 // Check convergence
159 residuals = std::abs(function_old);
160 step_norm = std::abs(step_old);
161 if (this->m_verbose) {this->info(residuals);}
162 if (residuals < tolerance_residuals || step_norm < tolerance_step_norm) {
163 this->m_converged = true;
164 break;
165 }
166
167 if (this->m_damped) {
168 // Relax the iteration process
169 damped = this->damp(std::forward<FunctionLambda>(function), x_old, function_old, step_old, x_new, function_new, step_new);
170 OPTIMIST_ASSERT_WARNING(damped, CMD "damping failed.");
171 } else {
172 // Update point
173 x_new = x_old + step_old;
174 success = this->evaluate_function(std::forward<FunctionLambda>(function), x_new, function_new);
175 OPTIMIST_ASSERT(success,
176 CMD "function evaluation failed at iteration " << this->m_iterations << ".");
177 }
178
179 // Update internal variables
180 x_old = x_new;
181 function_old = function_new;
182 step_old = step_new;
183 }
184
185 // Print bottom
186 if (this->m_verbose) {this->bottom();}
187
188 // Convergence data
189 x_sol = x_old;
190 return this->m_converged;
191
192 #undef CMD
193 }

205 template <typename FunctionLambda>
206 void compute_step(FunctionLambda && function, Scalar x_old, Scalar function_old, Scalar first_derivative_old,
207 Scalar & step_old)
208 {
209 #define CMD "Optimist::RootFinder::Varona::compute_step(...): "
210
211 bool success;
212 Scalar function_y, function_z, function_w, function_h, step_tmp, t, s, u, v;
213 Scalar tolerance_residuals{this->m_tolerance};
214
215 // Base step
216 step_old = -function_old/first_derivative_old;
217
218 // Order 4 step
219 if (this->m_method == Method::ORDER_4 || this->m_method == Method::ORDER_8 ||
220 this->m_method == Method::ORDER_16 || this->m_method == Method::ORDER_32) {
221 success = this->evaluate_function(std::forward<FunctionLambda>(function), x_old+step_old, function_y);
222 OPTIMIST_ASSERT(success,
223 CMD "function evaluation failed during order 4 step.");
224 if (std::abs(function_y) < tolerance_residuals) {return;}
225 t = function_y/function_old;
226 step_tmp = this->Q(t) * (function_y/first_derivative_old);
227 if (std::isfinite(step_tmp)) {step_old -= step_tmp;}
228 else {return;}
229 }
230
231 // Order 8 step (continued order 4)
232 if (this->m_method == Method::ORDER_8 || this->m_method == Method::ORDER_16 ||
233 this->m_method == Method::ORDER_32) {
234 success = this->evaluate_function(std::forward<FunctionLambda>(function), x_old+step_old, function_z);
235 OPTIMIST_ASSERT(success,
236 CMD "function evaluation failed during order 8 step.");
237 if (std::abs(function_z) < tolerance_residuals) {return;}
238 s = function_z/function_y;
239 step_tmp = this->W(t, s) * (function_z/first_derivative_old);
240 if (std::isfinite(step_tmp)) {step_old -= step_tmp;}
241 else {return;}
242 }
243
244 // Order 16 step (continued order 8)
245 if (this->m_method == Method::ORDER_16 || this->m_method == Method::ORDER_32) {
246 success = this->evaluate_function(std::forward<FunctionLambda>(function), x_old+step_old, function_w);
247 OPTIMIST_ASSERT(success,
248 CMD "function evaluation failed during order 16 step.");
249 if (std::abs(function_w) < tolerance_residuals) {return;}
250 u = function_w/function_z;
251 step_tmp = this->H(t, s, u) * (function_w/first_derivative_old);
252 if (std::isfinite(step_tmp)) {step_old -= step_tmp;}
253 else {return;}
254 }
255
256 // Order 32 step (continued order 16)
257 if (this->m_method == Method::ORDER_32) {
258 success = this->evaluate_function(std::forward<FunctionLambda>(function), x_old+step_old, function_h);
259 OPTIMIST_ASSERT(success,
260 CMD "function evaluation failed during order 32 step.");
261 if (std::abs(function_h) < tolerance_residuals) {return;}
262 v = (function_h/function_w);
263 step_tmp = this->J(t, s, u, v) * (function_h/first_derivative_old);
264 if (std::isfinite(step_tmp)) {step_old -= step_tmp;}
265 else {return;}
266 }
267
268 OPTIMIST_ASSERT(std::isfinite(step_old),
269 CMD "step is not finite.");
270
271 #undef CMD
272 }

279 static Scalar Q(Scalar t) {return 1.0 + 2.0*t;}

287 static Scalar W(Scalar t, Scalar s) {return t*t*(1.0 - 4.0*t) + (4.0*s + 2.0)*t + s + 1.0;}
288
296 static Scalar H(Scalar t, Scalar s, Scalar u) {
297 Scalar t1{t*t};
298 Scalar t2{t1*t1};
299 Scalar t8{s*s};
300 Scalar t17{s*t8};
301 Scalar t23{static_cast<Scalar>(2.0)*u};
302 return
303 ((8.0*u + 6.0*t2 + 4.0)*s -
304 (6.0*t8 + 4.0*(s + u + 1.0))*t1 +
305 2.0*t8 - 4.0*t17 + t23 + 2.0)*t +
306 t1*(t8 + s + u + 1.0) +
307 (1.0 - 3.0*t2 + t23)*s +
308 u - t17 + 1.0;
309 }
310
319 static Scalar J(Scalar t, Scalar s, Scalar u, Scalar v) {
320 Scalar t1{s*s};
321 Scalar t2{t1*t1};
322 Scalar t17{t*t};
323 Scalar t22{u*u};
324 Scalar t32{t17*t17};
325 Scalar t34{t*t32};
326 Scalar t37{t*t17};
327 Scalar t46{static_cast<Scalar>(1.0) + v};
328 Scalar t65{u + static_cast<Scalar>(1.0) + v};
329 Scalar t76{static_cast<Scalar>(-2.0*t22 + u + 4.0*v + 2.0)*u};
330 return
331 (-1.0 + 2.0*t)*(2.0 + 5.0*t)*u*t*t2 +
332 (4.0*t + 1.0)*u*s*t2 +
333 (u*t22 - 2.0*u*v - u - v - 1.0)*(4.0*t17 + 3.0*t + 1.0)*( - 1.0 + t) -
334 8.0*(t22*(t17/2.0 - 1.0/4.0) + u*(t17*t32 - 5.0/8.0*t34 - 3.0/4.0*t32 +
335 3.0/8.0*t37 + 3.0/4.0*t17 - t/8.0 - 1.0/4.0) + 3.0/4.0*t46*(t + 1.0/2.0)*(t - 2.0/3.0))*t*t1 +
336 4.0*(t22*( - 3.0/2.0*t - 1.0/4.0) + u*(t34 - t32 - 3.0/2.0*t37 + t17/4.0 - t - 1.0/4.0) -
337 t46*(t + 1.0/4.0))*s*t1 + (1.0 + v + t65*t17 - 4.0*t65*t37 - 3.0*t65*t32 + 6.0*t65*t34 +
338 t76 + 4.0*(1.0 + v + t76)*t)*s;
339 }
