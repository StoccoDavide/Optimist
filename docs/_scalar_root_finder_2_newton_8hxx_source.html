66 bool solve_impl(FunctionWrapper function, FirstDerivativeWrapper first_derivative, Real x_ini,
67 Real &x_sol)
68 {
69 #define CMD "Optimist::ScalarRootFinder::Newton::solve(...): "
70
71 // Setup internal variables
72 this->reset();
73
74 // Print header
75 if (this->m_verbose) {this->header();}
76
77 // Initialize variables
78 bool damped;
79 Real residuals, step_norm;
80 Real x_old, x_new, function_old, function_new, step_old, step_new;
81 Real first_derivative_old;
82
83 // Set initial iteration
84 x_old = x_ini;
85 this->evaluate_function(function, x_old, function_old);
86
87 // Algorithm iterations
88 Real tolerance_residuals{this->m_tolerance};
89 Real tolerance_step_norm{this->m_tolerance * this->m_tolerance};
90 for (this->m_iterations = Integer(1); this->m_iterations < this->m_max_iterations; ++this->m_iterations)
91 {
92 // Store trace
93 this->evaluate_first_derivative(first_derivative, x_old, first_derivative_old);
94 this->store_trace(x_old);
95
96 // Calculate step
97 if (std::abs(first_derivative_old) < EPSILON_LOW) {
98 OPTIMIST_WARNING( CMD "singular first derivative detected.");
99 first_derivative_old = (first_derivative_old > Real(0.0)) ? EPSILON_LOW : -EPSILON_LOW;
100 }
101 step_old = -function_old/first_derivative_old;
102
103 // Check convergence
104 residuals = std::abs(function_old);
105 step_norm = std::abs(step_old);
106 if (this->m_verbose) {this->info(residuals);}
107 if (residuals < tolerance_residuals || step_norm < tolerance_step_norm) {
108 this->m_converged = true;
109 break;
110 }
111
112 if (this->m_damped)
113 {
114 // Relax the iteration process
115 damped = this->damp(function, x_old, function_old, step_old, x_new, function_new, step_new);
116 OPTIMIST_ASSERT_WARNING(damped, CMD "damping failed.");
117 } else {
118 // Update point
119 x_new = x_old + step_old;
120 this->evaluate_function(function, x_new, function_new);
121 }
122
123 // Update internal variables
124 x_old = x_new;
125 function_old = function_new;
126 step_old = step_new;
127 }
128
129 // Print bottom
130 if (this->m_verbose) {this->bottom();}
131
132 // Convergence data
133 x_sol = x_old;
134 return this->m_converged;
135
136 #undef CMD
137 }
