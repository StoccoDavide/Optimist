83 bool solve(Vector const &x_ini, Vector &x_sol) override
84 {
85 // Setup internal variables
86 this->reset();
87
88 // Print header
89 if (this->m_verbose) {this->header();}
90
91 // Initialize variables
92 Real residuals, step_norm;
93 Vector x_old, x_new, function_old, function_new, step_old, step_new;
94 Matrix jacobian;
95
96 // Set initial iteration
97 x_old = x_ini;
98 this->evaluate_function(x_old, function_old);
99
100 // Algorithm iterations
101 Real tolerance_residuals{this->m_tolerance};
102 Real tolerance_step_norm{this->m_tolerance * this->m_tolerance};
103 for (this->m_iterations = Integer(1); this->m_iterations < this->m_max_iterations; ++this->m_iterations)
104 {
105 // Store trace
106 this->store_trace(x_old);
107
108 // Calculate step
109 this->evaluate_jacobian(x_old, jacobian);
110 this->m_lu.compute(jacobian);
111 OPTIMIST_ASSERT(this->m_lu.rank() == N,
112 "Optimist::RootFinder::Newton::solve(...): singular Jacobian detected.");
113 step_old = -this->m_lu.solve(function_old);
114
115 // Check convergence
116 residuals = function_old.norm();
117 step_norm = step_old.norm();
118 if (this->m_verbose) {this->info(residuals);}
119 if (residuals < tolerance_residuals || step_norm < tolerance_step_norm) {
120 this->m_converged = true;
121 break;
122 }
123
124 if (this->m_damped)
125 {
126 // Relax the iteration process
127 bool damped{this->damp(x_old, function_old, step_old, x_new, function_new, step_new)};
128 OPTIMIST_ASSERT_WARNING(damped,
129 "Optimist::RootFinder::Newton::solve(...): damping failed.");
130 } else {
131 // Update point
132 x_new = x_old + step_old;
133 this->evaluate_function(x_new, function_new);
134 }
135
136 // Update internal variables
137 x_old = x_new;
138 function_old = function_new;
139 step_old = step_new;
140 }
141
142 // Print bottom
143 if (this->m_verbose) {this->bottom();}
144
145 // Convergence data
146 x_sol = x_old;
147 return this->m_converged;
148 }
