74 bool solve_impl(Function t_function, Jacobian t_jacobian, Vector const &x_ini, Vector &x_sol)
75 {
76 // Setup internal variables
77 this->reset();
78
79 // Print header
80 if (this->m_verbose) {this->header();}
81
82 // Initialize variables
83 bool damped;
84 Real residuals, step_norm;
85 Vector x_old, x_new, function_old, function_new, step_old, step_new;
86 Matrix jacobian;
87
88 // Set initial iteration
89 x_old = x_ini;
90 this->evaluate_function(t_function, x_old, function_old);
91
92 // Algorithm iterations
93 Real tolerance_residuals{this->m_tolerance};
94 Real tolerance_step_norm{this->m_tolerance * this->m_tolerance};
95 for (this->m_iterations = Integer(1); this->m_iterations < this->m_max_iterations; ++this->m_iterations)
96 {
97 // Store trace
98 this->store_trace(x_old);
99
100 // Calculate step
101 this->evaluate_jacobian(t_jacobian, x_old, jacobian);
102 this->m_lu.compute(jacobian);
103 OPTIMIST_ASSERT(this->m_lu.rank() == N,
104 "Optimist::RootFinder::Newton::solve(...): singular Jacobian detected.");
105 step_old = -this->m_lu.solve(function_old);
106
107 // Check convergence
108 residuals = function_old.norm();
109 step_norm = step_old.norm();
110 if (this->m_verbose) {this->info(residuals);}
111 if (residuals < tolerance_residuals || step_norm < tolerance_step_norm) {
112 this->m_converged = true;
113 break;
114 }
115
116 if (this->m_damped)
117 {
118 // Relax the iteration process
119 damped = this->damp(t_function, x_old, function_old, step_old, x_new, function_new, step_new);
120 OPTIMIST_ASSERT_WARNING(damped,
121 "Optimist::RootFinder::Newton::solve(...): damping failed.");
122 } else {
123 // Update point
124 x_new = x_old + step_old;
125 this->evaluate_function(t_function, x_new, function_new);
126 }
127
128 // Update internal variables
129 x_old = x_new;
130 function_old = function_new;
131 step_old = step_new;
132 }
133
134 // Print bottom
135 if (this->m_verbose) {this->bottom();}
136
137 // Convergence data
138 x_sol = x_old;
139 return this->m_converged;
140 }
