78 bool solve_impl(FunctionWrapper function, JacobianWrapper jacobian, Vector const & x_ini, Vector & x_sol)
79 {
80 // Setup internal variables
81 this->reset();
82
83 // Print header
84 if (this->m_verbose) {this->header();}
85
86 // Initialize variables
87 bool damped;
88 Real residuals, step_norm;
89 Vector x_old, x_new, function_old, function_new, step_old, step_new;
90 Matrix jacobian_old;
91
92 // Set initial iteration
93 x_old = x_ini;
94 this->evaluate_function(function, x_old, function_old);
95
96 // Algorithm iterations
97 Real tolerance_residuals{this->m_tolerance};
98 Real tolerance_step_norm{this->m_tolerance * this->m_tolerance};
99 for (this->m_iterations = Integer(1); this->m_iterations < this->m_max_iterations; ++this->m_iterations)
100 {
101 // Store trace
102 this->store_trace(x_old);
103
104 // Calculate step
105 this->evaluate_jacobian(jacobian, x_old, jacobian_old);
106 this->m_lu.compute(jacobian_old);
107 OPTIMIST_ASSERT(this->m_lu.rank() == N,
108 "Optimist::RootFinder::Newton::solve(...): singular Jacobian detected.");
109 step_old = -this->m_lu.solve(function_old);
110
111 // Check convergence
112 residuals = function_old.norm();
113 step_norm = step_old.norm();
114 if (this->m_verbose) {this->info(residuals);}
115 if (residuals < tolerance_residuals || step_norm < tolerance_step_norm) {
116 this->m_converged = true;
117 break;
118 }
119
120 if (this->m_damped)
121 {
122 // Relax the iteration process
123 damped = this->damp(function, x_old, function_old, step_old, x_new, function_new, step_new);
124 OPTIMIST_ASSERT_WARNING(damped,
125 "Optimist::RootFinder::Newton::solve(...): damping failed.");
126 } else {
127 // Update point
128 x_new = x_old + step_old;
129 this->evaluate_function(function, x_new, function_new);
130 }
131
132 // Update internal variables
133 x_old = x_new;
134 function_old = function_new;
135 step_old = step_new;
136 }
137
138 // Print bottom
139 if (this->m_verbose) {this->bottom();}
140
141 // Convergence data
142 x_sol = x_old;
143 return this->m_converged;
144 }
