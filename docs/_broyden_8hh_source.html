118 void update_impl(
119 Vector const & delta_x_old, Vector const & delta_function_old, FirstDerivative const & jacobian_old,
120 Vector const & delta_x_new, Vector const & delta_function_new, Vector const & /*function_new*/,
121 FirstDerivative & jacobian_new
122 ) {
123 Vector tmp_1(jacobian_old * delta_function_new);
124 Scalar tmp_2{delta_function_new.squaredNorm()};
125 // Selection criteria: |(dx_new'*dx_old) / (dx_new'*J_old*dF_new)| < |(dF_new'*dF_old) / (dF_new'*dF_new)|
126 if (this->m_method == Method::COMBINED || this->m_method == Method::GOOD || this->iterations() < Integer(2) ||
127 std::abs(delta_x_new.dot(delta_x_old)) / std::abs(delta_x_new.dot(tmp_1))
128 < std::abs(delta_function_new.dot(delta_function_old)) / tmp_2) {
129 // Broyden's Good solver: J_new = J_old - (J_old*dF_new-dx_new) / (C'*dF_new)*C', where C = J_old'*dx_new;
130 Vector C_g(jacobian_old.transpose() * delta_x_new);
131 jacobian_new = jacobian_old - (tmp_1 - delta_x_new) / (C_g.dot(delta_function_new)) * C_g.transpose();
132 } else {
133 // Broyden's Bad solver: J_new = J_old - (J_old*dF_old-dx_new) / (C'*dF_old)*C', where C = dF_old;
134 jacobian_new = jacobian_old - (tmp_1 - delta_x_new) / tmp_2 * delta_function_old.transpose();
135 }
136 }
