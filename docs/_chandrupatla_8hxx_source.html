67 Real find_root_impl(FunctionWrapper function) {
68
69 #define CMD "Optimist::ScalarRootfinder::Chandrupatla::find_root_impl(...): "
70
71 Real tolerance_step{this->m_tolerance_bracketing};
72 Real tolerance_function{this->m_tolerance_bracketing};
73
74 // Check for trivial solution
75 this->m_converged = this->m_fa == 0; if (this->m_converged) {return this->m_a;}
76 this->m_converged = this->m_fb == 0; if (this->m_converged) {return this->m_b;}
77
78 // Initialize to dumb values
79 this->m_e = QUIET_NAN;
80 this->m_fe = QUIET_NAN;
81
82 // While f(left) or f(right) are infinite perform bisection
83 while (!(std::isfinite(m_fa) && std::isfinite(m_fb))) {
84 ++this->m_iterations;
85 this->m_c = (this->m_a + this->m_b)/2.0;
86 this->evaluate_function(function, this->m_c, this->m_fc);
87 this->m_converged = this->m_fc == 0;
88 if (this->m_converged) {return this->m_c;}
89 if (this->m_fa*this->m_fc < 0.0) {
90 // -> [a, c]
91 this->m_b = this->m_c; this->m_fb = this->m_fc;
92 } else {
93 // -> [c, b]
94 this->m_a = this->m_c; this->m_fa = this->m_fc;
95 }
96
97 // Check for convergence
98 Real abs_fa{std::abs(this->m_fa)};
99 Real abs_fb{std::abs(this->m_fb)};
100 this->m_converged = (this->m_b - this->m_a) < tolerance_step ||
101 abs_fa < tolerance_function ||abs_fb < tolerance_function;
102 if (this->m_converged) {return abs_fb < abs_fa ? this->m_b : this->m_a;}
103 }
104
105 Real t{0.5};
106
107 while (++this->m_iterations < this->m_max_iterations) {
108
109 Real direction{this->m_b - this->m_a};
110 Real tolerance{tolerance_step/(2.0*std::abs(direction))};
111 this->m_converged = tolerance >= 0.5;
112 if (this->m_converged) {break;}
113
114 Real abs_fa{std::abs(this->m_fa)};
115 Real abs_fb{std::abs(this->m_fb)};
116 this->m_converged = abs_fa < tolerance_function || abs_fb < tolerance_function;
117 if (this->m_converged) {break;}
118
119 if (t < tolerance) {t = tolerance;}
120 else if (t > 1.0 - tolerance) {t = 1.0 - tolerance;}
121
122 Real c{this->m_a + t * direction};
123 Real fc; this->evaluate_function(function, c, fc);
124 if (this->m_verbose) {this->info(fc);}
125
126 OPTIMIST_ASSERT(std::isfinite(fc), CMD "function is not finite.");
127
128 this->m_converged = fc == 0;
129 if (this->m_converged) {
130 this->m_a = this->m_b = c; this->m_fa = this->m_fb = 0;
131 break;
132 }
133
134 // Arrange 2-1-3: 2-1 interval, 1 middle, 3 discarded point
135 Real d, fd;
136 if ((0.0 < fc) == (0.0 < this->m_fa)) {
137 // a -> d --> [d,a,b] = [a,c,b]
138 d = this->m_a; fd = this->m_fa;
139 } else {
140 // b -> d, a -> b --> [b,a,d] = [a,c,b]
141 d = this->m_b; fd = this->m_fb;
142 this->m_b = this->m_a; this->m_fb = this->m_fa;
143 }
144 // c => a
145 this->m_a = c; this->m_fa = fc;
146
147 // If inverse quadratic interpolation holds use it
148 Real ba{this->m_b - this->m_a};
149 Real fba{this->m_fb - this->m_fa};
150 Real bd{this->m_b - d};
151 Real fbd{this->m_fb - fd};
152
153 Real xi{ba/bd};
154 Real ph{fba/fbd};
155 Real fl{1.0 - std::sqrt(1.0 - xi)};
156 Real fh{std::sqrt(xi)};
157
158 if (fl < ph && ph < fh) {
159 Real da{d - this->m_a};
160 Real fda{fd - this->m_fa};
161 t = (this->m_fa/fba) * (fd/fbd) - (this->m_fa/fda) * (this->m_fb/fbd) * (da/ba);
162 } else {
163 t = 0.5;
164 }
165 }
166 if (this->m_a > this->m_b) {std::swap(this->m_a, this->m_b); std::swap(this->m_fa, this->m_fb);}
167 return std::abs(this->m_fa) < std::abs(this->m_fb) ? this->m_a : this->m_b;;
168
169 #undef CMD
170 }
