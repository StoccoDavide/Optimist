71 Real find_root_impl(FunctionWrapper function) {
72
73 #define CMD "Optimist::ScalarRootfinder::Chandrupatla::find_root_impl(...): "
74
75 Real tolerance_step{this->m_tolerance_bracketing};
76 Real tolerance_function{this->m_tolerance_bracketing};
77
78 // Check for trivial solution
79 this->m_converged = this->m_fa == 0; if (this->m_converged) {return this->m_a;}
80 this->m_converged = this->m_fb == 0; if (this->m_converged) {return this->m_b;}
81
82 // Initialize to dumb values
83 this->m_e = QUIET_NAN;
84 this->m_fe = QUIET_NAN;
85
86 // While f(left) or f(right) are infinite perform bisection
87 while (!(std::isfinite(this->m_fa) && std::isfinite(this->m_fb))) {
88 ++this->m_iterations;
89 this->m_c = (this->m_a + this->m_b)/2.0;
90 this->evaluate_function(function, this->m_c, this->m_fc);
91 this->m_converged = this->m_fc == 0;
92 if (this->m_converged) {return this->m_c;}
93 if (this->m_fa*this->m_fc < 0.0) {
94 // -> [a, c]
95 this->m_b = this->m_c; this->m_fb = this->m_fc;
96 } else {
97 // -> [c, b]
98 this->m_a = this->m_c; this->m_fa = this->m_fc;
99 }
100
101 // Check for convergence
102 Real abs_fa{std::abs(this->m_fa)};
103 Real abs_fb{std::abs(this->m_fb)};
104 this->m_converged = (this->m_b - this->m_a) < tolerance_step ||
105 abs_fa < tolerance_function ||abs_fb < tolerance_function;
106 if (this->m_converged) {return abs_fb < abs_fa ? this->m_b : this->m_a;}
107 }
108
109 Real t{0.5};
110
111 while (++this->m_iterations < this->m_max_iterations) {
112
113 Real direction{this->m_b - this->m_a};
114 Real tolerance{tolerance_step/(2.0*std::abs(direction))};
115 this->m_converged = tolerance >= 0.5;
116 if (this->m_converged) {break;}
117
118 Real abs_fa{std::abs(this->m_fa)};
119 Real abs_fb{std::abs(this->m_fb)};
120 this->m_converged = abs_fa < tolerance_function || abs_fb < tolerance_function;
121 if (this->m_converged) {break;}
122
123 if (t < tolerance) {t = tolerance;}
124 else if (t > 1.0 - tolerance) {t = 1.0 - tolerance;}
125
126 Real c{this->m_a + t * direction};
127 Real fc; this->evaluate_function(function, c, fc);
128 if (this->m_verbose) {this->info(fc);}
129
130 OPTIMIST_ASSERT(std::isfinite(fc), CMD "function is not finite.");
131
132 this->m_converged = fc == 0;
133 if (this->m_converged) {
134 this->m_a = this->m_b = c; this->m_fa = this->m_fb = 0;
135 break;
136 }
137
138 // Arrange 2-1-3: 2-1 interval, 1 middle, 3 discarded point
139 Real d, fd;
140 if ((0.0 < fc) == (0.0 < this->m_fa)) {
141 // a -> d --> [d,a,b] = [a,c,b]
142 d = this->m_a; fd = this->m_fa;
143 } else {
144 // b -> d, a -> b --> [b,a,d] = [a,c,b]
145 d = this->m_b; fd = this->m_fb;
146 this->m_b = this->m_a; this->m_fb = this->m_fa;
147 }
148 // c => a
149 this->m_a = c; this->m_fa = fc;
150
151 // If inverse quadratic interpolation holds use it
152 Real ba{this->m_b - this->m_a};
153 Real fba{this->m_fb - this->m_fa};
154 Real bd{this->m_b - d};
155 Real fbd{this->m_fb - fd};
156
157 Real xi{ba/bd};
158 Real ph{fba/fbd};
159 Real fl{1.0 - std::sqrt(1.0 - xi)};
160 Real fh{std::sqrt(xi)};
161
162 if (fl < ph && ph < fh) {
163 Real da{d - this->m_a};
164 Real fda{fd - this->m_fa};
165 t = (this->m_fa/fba) * (fd/fbd) - (this->m_fa/fda) * (this->m_fb/fbd) * (da/ba);
166 } else {
167 t = 0.5;
168 }
169 }
170 if (this->m_a > this->m_b) {std::swap(this->m_a, this->m_b); std::swap(this->m_fa, this->m_fb);}
171 return std::abs(this->m_fa) < std::abs(this->m_fb) ? this->m_a : this->m_b;;
172
173 #undef CMD
174 }
