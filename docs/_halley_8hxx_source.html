72 bool solve_impl(FunctionWrapper function, FirstDerivativeWrapper first_derivative,
73 SecondDerivativeWrapper second_derivative, Real x_ini, Real & x_sol)
74 {
75 #define CMD "Optimist::ScalarRootFinder::Halley::solve(...): "
76
77 // Setup internal variables
78 this->reset();
79
80 // Print header
81 if (this->m_verbose) {this->header();}
82
83 // Initialize variables
84 bool damped;
85 Real residuals, step_norm;
86 Real x_old, x_new, function_old, function_new, step_old, step_new;
87 Real first_derivative_old, second_derivative_old;
88
89 // Set initial iteration
90 x_old = x_ini;
91 this->evaluate_function(function, x_old, function_old);
92
93 // Algorithm iterations
94 Real tolerance_residuals{this->m_tolerance};
95 Real tolerance_step_norm{this->m_tolerance * this->m_tolerance};
96 for (this->m_iterations = Integer(1); this->m_iterations < this->m_max_iterations; ++this->m_iterations)
97 {
98 // Store trace
99 this->store_trace(x_old);
100
101 // Evaluate derivatives
102 this->evaluate_first_derivative(first_derivative, x_old, first_derivative_old);
103 this->evaluate_second_derivative(second_derivative, x_old, second_derivative_old);
104
105 // Calculate step
106 if (std::abs(first_derivative_old) < EPSILON_LOW) {
107 OPTIMIST_WARNING( CMD "singular first derivative detected.");
108 first_derivative_old = (first_derivative_old > 0.0) ? EPSILON_LOW : -EPSILON_LOW;
109 }
110 if (std::abs(second_derivative_old) < EPSILON_LOW) {
111 OPTIMIST_WARNING( CMD "singular second derivative detected.");
112 second_derivative_old = (second_derivative_old > 0.0) ? EPSILON_LOW : -EPSILON_LOW;
113 }
114 step_old = -(function_old / first_derivative_old) * (1.0 - (function_old * second_derivative_old) /
115 (first_derivative_old * first_derivative_old));
116 OPTIMIST_ASSERT(std::isfinite(step_old), CMD "step " << this->m_iterations << " is not finite.");
117
118 // Check convergence
119 residuals = std::abs(function_old);
120 step_norm = std::abs(step_old);
121 if (this->m_verbose) {this->info(residuals);}
122 if (residuals < tolerance_residuals || step_norm < tolerance_step_norm) {
123 this->m_converged = true;
124 break;
125 }
126
127 if (this->m_damped) {
128 // Relax the iteration process
129 damped = this->damp(function, x_old, function_old, step_old, x_new, function_new, step_new);
130 OPTIMIST_ASSERT_WARNING(damped, CMD "damping failed.");
131 } else {
132 // Update point
133 x_new = x_old + step_old;
134 this->evaluate_function(function, x_new, function_new);
135 }
136
137 // Update internal variables
138 x_old = x_new;
139 function_old = function_new;
140 step_old = step_new;
141 }
142
143 // Print bottom
144 if (this->m_verbose) {this->bottom();}
145
146 // Convergence data
147 x_sol = x_old;
148 return this->m_converged;
149
150 #undef CMD
151 }
